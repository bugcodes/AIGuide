### 10.1 斜杠命令概述

#### 什么是斜杠命令？

```
斜杠命令 = /command [args]

示例：
/git-commit "添加用户登录功能"
/review src/auth.py --focus=security
/test-gen src/utils/helper.py --framework=jest
```

#### 斜杠命令 vs 普通对话

| 维度 | 斜杠命令 | 普通对话 |
|------|---------|---------|
| **交互方式** | 结构化输入 | 自然语言 |
| **参数明确性** | 参数清晰 | 需要理解意图 |
| **可重复性** | 高 | 中 |
| **自动化程度** | 高 | 低 |

### 10.2 斜杠命令解析器

```python
import re
from dataclasses import dataclass
from typing import List, Optional, Dict, Any


@dataclass
class ParsedCommand:
    """解析后的命令"""
    name: str
    args: List[str]
    flags: Dict[str, str]  # --flag=value
    positional: List[str]  # 位置参数


class SlashCommandParser:
    """斜杠命令解析器"""

    def __init__(self):
        self.commands: Dict[str, Any] = {}

    def register(self, name: str, handler: callable, schema: Dict = None):
        """注册命令"""
        self.commands[name] = {
            "handler": handler,
            "schema": schema or {}
        }

    def parse(self, input_str: str) -> ParsedCommand:
        """
        解析命令字符串

        Examples:
            /git-commit "feat: add login"
            /review src/ --focus=security --verbose
            /test-gen user.py --framework=pytest --coverage
        """
        # 移除开头的 /
        input_str = input_str.lstrip('/')

        # 分割命令名称和参数
        parts = input_str.split()

        if not parts:
            raise ValueError("Empty command")

        name = parts[0]
        args_str = ' '.join(parts[1:]) if len(parts) > 1 else ""

        # 解析参数
        args, flags, positional = self._parse_args(args_str)

        return ParsedCommand(
            name=name,
            args=args,
            flags=flags,
            positional=positional
        )

    def _parse_args(self, args_str: str) -> tuple:
        """解析参数字符串"""
        args = []
        flags = {}
        positional = []

        if not args_str.strip():
            return args, flags, positional

        # 使用正则解析
        # 匹配 --flag=value 或 --flag value 格式
        flag_pattern = r'--([a-zA-Z0-9_-]+)(?:=(.+?))?'

        remaining = args_str

        # 解析命名参数
        for match in re.finditer(flag_pattern, args_str):
            flag_name = match.group(1)
            flag_value = match.group(2) if match.group(2) else "true"
            flags[flag_name] = flag_value

        # 移除已解析的参数
        remaining = re.sub(flag_pattern, '', args_str)

        # 解析位置参数（引号内的内容）
        positional_pattern = r'"([^"]+)"|\'([^\']+)\'|(\S+)'
        for match in re.finditer(positional_pattern, remaining):
            value = match.group(1) or match.group(2) or match.group(3)
            positional.append(value.strip())

        return args, flags, positional

    async def execute(self, input_str: str) -> Any:
        """执行命令"""
        parsed = self.parse(input_str)

        if parsed.name not in self.commands:
            raise ValueError(f"Unknown command: {parsed.name}")

        command = self.commands[parsed.name]

        # 验证参数
        if command["schema"]:
            self._validate(parsed, command["schema"])

        # 执行
        handler = command["handler"]
        result = handler(parsed.args, parsed.flags, parsed.positional)

        if hasattr(result, '__anext__') or hasattr(result, '__await__'):
            result = await result

        return result

    def _validate(self, parsed: ParsedCommand, schema: Dict):
        """验证参数"""
        # 检查必需参数
        required = schema.get("required", [])
        for req in required:
            if req not in parsed.flags and req not in parsed.positional:
                raise ValueError(f"Missing required argument: {req}")

        # 检查参数类型
        params = schema.get("params", {})
        for name, param_type in params.items():
            if name in parsed.flags:
                value = parsed.flags[name]
                if not self._check_type(value, param_type):
                    raise ValueError(f"Invalid type for {name}: expected {param_type}")
```

### 10.3 任务型 Skill 示例

#### 示例：Git Release Skill

```markdown
---
name: git-release
description: "自动化 Git Release 流程"
version: "1.0.0"
author: "Claude Team"
tags: ["git", "release", "ci-cd"]
trigger: "/release"
---

# Git Release Skill

## 描述

自动化 Git Release 流程，包括：
1. 版本号检查和更新
2. 生成 CHANGELOG
3. 创建 Git Tag
4. 推送标签触发 CI/CD

## 使用方法

```
/release [version] [options]

参数：
  version    版本号（如 1.2.3）或 'patch'/'minor'/'major'

选项：
  --dry-run    模拟运行，不实际执行
  --skip-tests 跳过测试
  --force      强制执行
  --message    自定义提交信息
```

## 完整流程

### Step 1: 版本检查
- 检查当前版本（从 package.json 或 VERSION 文件）
- 验证新版本号格式
- 检查是否有未提交的更改

### Step 2: 生成变更日志
- 从 git log 提取提交信息
- 按类型分类（feat, fix, docs 等）
- 生成 CHANGELOG.md

### Step 3: 版本更新
- 更新 package.json 或 VERSION 文件
- 提交更改
- 创建 Git Tag

### Step 4: 推送
- 推送提交
- 推送标签
- 触发 CI/CD 流程
```

```
## 错误处理

| 错误 | 原因 | 解决方案 |
|------|------|----------|
| 未提交更改 | 存在未提交的更改 | 先提交或使用 --force |
| 无效版本号 | 版本号格式错误 | 使用 semver 格式 |
| 远程有更新 | 远程分支有新的提交 | 先 pull 或 rebase |
| Tag 已存在 | Tag 已被使用 | 使用 --force 或新版本 |

## 注意事项

1. 需要 Git 仓库已初始化
2. 需要 push 权限
3. 建议先使用 --dry-run 检查
4. 建议确保测试通过后再 release
```

### 10.4 交互式任务 Skill

```python
class InteractiveSkill:
    """交互式 Skill"""

    def __init__(self):
        self.state: Dict[str, Any] = {}
        self.current_step: int = 0
        self.steps: List[Step] = []

    async def run(self, initial_input: str = None):
        """运行交互式 Skill"""
        if initial_input:
            self.state["initial_input"] = initial_input
            return await self._execute_step(0)

        return "Waiting for user input"

    async def _execute_step(self, step_index: int):
        """执行指定步骤"""
        if step_index >= len(self.steps):
            return await self._complete()

        step = self.steps[step_index]
        self.current_step = step_index

        # 执行步骤
        result = await step.execute(self.state)

        # 处理结果
        if isinstance(result, StepResult):
            if result.next_step:
                return await self._execute_step(result.next_step)
            elif result.wait_for_input:
                self.state["pending_step"] = step_index
                return result.message
            elif result.complete:
                return await self._complete(result.data)

        return result


@dataclass
class Step:
    """交互步骤"""
    name: str
    prompt: str
    validate: Callable[[str], bool]
    process: Callable[[str, Dict], Any]
    next_step: Optional[int] = None


@dataclass
class StepResult:
    """步骤结果"""
    message: str = ""
    next_step: Optional[int] = None
    wait_for_input: bool = False
    complete: bool = False
    data: Any = None


# 示例：用户创建 Skill
class CreateProjectSkill(InteractiveSkill):
    """创建项目交互式 Skill"""

    def __init__(self):
        super().__init__()
        self.steps = [
            Step(
                name="project_name",
                prompt="请输入项目名称：",
                validate=self._validate_name,
                process=self._process_name,
                next_step=1
            ),
            Step(
                name="project_type",
                prompt="请选择项目类型（1: Web, 2: API, 3: CLI）：",
                validate=self._validate_type,
                process=self._process_type,
                next_step=2
            ),
            Step(
                name="confirm",
                prompt="确认创建项目？",
                validate=self._validate_confirm,
                process=self._process_confirm,
                next_step=None
            )
        ]

    def _validate_name(self, name: str) -> bool:
        """验证项目名称"""
        if not name:
            return False
        if len(name) < 3:
            return False
        return True

    def _process_name(self, name: str, state: Dict) -> Any:
        """处理项目名称"""
        state["project_name"] = name
        return StepResult(message=f"项目名称: {name}", next_step=1)

    def _validate_type(self, input_str: str) -> bool:
        """验证项目类型"""
        return input_str in ["1", "2", "3"]

    def _process_type(self, input_str: str, state: Dict) -> Any:
        """处理项目类型"""
        types = {"1": "web", "2": "api", "3": "cli"}
        state["project_type"] = types[input_str]
        return StepResult(message=f"项目类型: {state['project_type']}", next_step=2)

    async def _complete(self, data: Any = None) -> str:
        """完成"""
        return f"项目创建完成: {self.state}"
```

### 10.5 参数验证与错误处理

```python
class ValidationError(Exception):
    """验证错误"""
    def __init__(self, field: str, message: str):
        self.field = field
        self.message = message
        super().__init__(f"{field}: {message}")


class CommandValidator:
    """命令验证器"""

    def __init__(self):
        self.rules: Dict[str, List[Callable]] = {}

    def add_rule(self, field: str, rule: Callable):
        """添加验证规则"""
        if field not in self.rules:
            self.rules[field] = []
        self.rules[field].append(rule)

    def validate(self, data: Dict) -> List[ValidationError]:
        """验证数据"""
        errors = []

        for field, rules in self.rules.items():
            value = data.get(field)

            for rule in rules:
                try:
                    rule(value)
                except ValidationError as e:
                    errors.append(e)

        return errors

    def validate_or_raise(self, data: Dict):
        """验证或抛出异常"""
        errors = self.validate(data)

        if errors:
            raise ValidationError(
                field=",
                message="\n".join(str(e) for e in errors)
            )


# 常用验证规则
class Rules:
    """验证规则"""

    @staticmethod
    def required(value) -> bool:
        """必填"""
        if value is None or value == "":
            raise ValidationError("required", "This field is required")
        return True

    @staticmethod
    def min_length(min_len: int):
        """最小长度"""
        def rule(value):
            if value and len(str(value)) < min_len:
                raise ValidationError(
                    "min_length",
                    f"Must be at least {min_len} characters"
                )
        return rule

    @staticmethod
    def max_length(max_len: int):
        """最大长度"""
        def rule(value):
            if value and len(str(value)) > max_len:
                raise ValidationError(
                    "max_length",
                    f"Must be at most {max_len} characters"
                )
        return rule

    @staticmethod
    def pattern(regex: str):
        """正则匹配"""
        def rule(value):
            if value and not re.match(regex, str(value)):
                raise ValidationError(
                    "pattern",
                    f"Does not match required pattern"
                )
        return rule

    @staticmethod
    def choices(*options):
        """可选值"""
        def rule(value):
            if value and value not in options:
                raise ValidationError(
                    "choices",
                    f"Must be one of: {', '.join(options)}"
                )
        return rule
```

### 10.6 任务模板系统

```python
class TaskTemplate:
    """任务模板"""

    def __init__(
        self,
        name: str,
        description: str,
        steps: List[Dict],
        parameters: Dict = None
    ):
        self.name = name
        self.description = description
        self.steps = steps
        self.parameters = parameters or {}

    def render(self, context: Dict) -> "RenderedTask":
        """渲染模板"""
        rendered_steps = []

        for step in self.steps:
            rendered_step = {
                "name": self._interpolate(step["name"], context),
                "command": self._interpolate(step["command"], context),
                "description": self._interpolate(step.get("description", ""), context)
            }
            rendered_steps.append(rendered_step)

        return RenderedTask(
            name=self.name,
            steps=rendered_steps,
            context=context
        )

    def _interpolate(self, template: str, context: Dict) -> str:
        """字符串插值"""
        result = template

        for key, value in context.items():
            result = result.replace(f"{{{key}}}", str(value))

        return result


class RenderedTask:
    """渲染后的任务"""

    def __init__(self, name: str, steps: List[Dict], context: Dict):
        self.name = name
        self.steps = steps
        self.context = context

    async def execute(self) -> Dict:
        """执行任务"""
        results = []

        for step in self.steps:
            result = await self._execute_step(step)
            results.append(result)

        return {
            "task": self.name,
            "steps": results,
            "success": all(r.get("success", False) for r in results)
        }

    async def _execute_step(self, step: Dict) -> Dict:
        """执行单个步骤"""
        try:
            # 这里应该是实际的命令执行
            # 简化示例
            return {
                "step": step["name"],
                "success": True,
                "output": f"Executed: {step['command']}"
            }
        except Exception as e:
            return {
                "step": step["name"],
                "success": False,
                "error": str(e)
            }
```
