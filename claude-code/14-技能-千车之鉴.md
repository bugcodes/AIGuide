### 12.1 Claude Code 常见错误

#### 错误分类概览

```
Claude Code 常见错误分类：

1. 配置错误
   - API 密钥配置错误
   - 权限设置不当
   - 环境变量缺失

2. 使用错误
   - 指令不明确
   - 上下文溢出
   - 过度依赖 AI

3. 安全错误
   - 敏感信息泄露
   - 不安全的代码生成
   - 权限过度开放

4. 协作错误
   - 任务分配不当
   - 上下文冲突
   - 同步问题
```

#### 配置错误详解

```python
# 错误1：API 密钥配置错误

# 错误示例
CONFIG_EXAMPLE_1 = {
    "api_key": "sk-xxxx",  # 暴露在代码中
    "provider": "openai",
    "model": "gpt-4"
}

# 正确示例
CONFIG_EXAMPLE_1_CORRECT = {
    "api_key": os.environ.get("OPENAI_API_KEY"),  # 从环境变量读取
    "provider": "openai",
    "model": "gpt-4"
}


# 错误2：权限过度开放

# 错误示例
PERMISSIONS_ERROR = {
    "dangerous": True,  # 允许删除文件
    "dangerous": True,  # 允许执行命令
    "dangerous": True,  # 允许网络访问
}

# 正确示例
PERMISSIONS_CORRECT = {
    "allow_destructive": False,
    "allow_exec": False,
    "allow_network": True,  # 仅允许必要的网络访问
}
```

### 12.2 Multi-Agent 系统常见错误

#### 错误模式

```python
# 模式1：循环依赖

# 错误示例
CIRCULAR_DEPENDENCY = {
    "Agent A": "需要 Agent B 的结果才能继续",
    "Agent B": "需要 Agent C 的结果才能继续",
    "Agent C": "需要 Agent A 的结果才能继续",

    # 结果：死锁，所有人都等待
}

# 解决方案：DAG 结构
DAG_SOLUTION = {
    "Agent A": {"depends_on": []},
    "Agent B": {"depends_on": ["Agent A"]},
    "Agent C": {"depends_on": ["Agent A"]},
    "Agent D": {"depends_on": ["Agent B", "Agent C"]},
    # 有向无环图，没有循环
}


# 模式2：上下文污染

# 错误示例
CONTEXT_POLLUTION = """
Agent 1: 输出了 5000 行的详细日志
Agent 2: 需要处理这些日志
Agent 3: 需要处理 Agent 2 的结果

问题：
- Agent 2 和 Agent 3 的上下文被污染
- 大量无关信息占用 token
- 分析效率下降
"""

# 解决方案：结果聚合
CONTEXT_POLLUTION_SOLUTION = """
# Agent 1 应该输出结构化摘要

Agent 1 输出：
{
    "summary": "发现 5 个 SQL 注入风险",
    "critical": 2,
    "high": 3,
    "files_affected": ["auth.py", "user.py"],
    "details": [...]
}

# 只有关键信息，不包含完整日志
"""


# 模式3：任务分配不均

# 错误示例
UNEVEN_DISTRIBUTION = """
Agent 1: 分配了 100 个小任务
Agent 2: 分配了 1 个大任务

结果：
- Agent 1 忙得不可开交
- Agent 2 空闲等待
- 整体效率低下
"""

# 解决方案：任务粒度均衡
BALANCED_DISTRIBUTION = """
任务分配策略：

1. 估算每个任务的复杂度
2. 将大任务拆分为子任务
3. 将小任务合并为批次
4. 确保每个 Agent 的工作量接近

示例：
- 将 100 个小任务打包成 4 个批次
- 每个 Agent 处理 1 个批次
- 或者每个 Agent 处理 25 个小任务
"""
```

#### 通信错误

```python
# 错误1：消息丢失

MESSAGE_LOSS = """
问题：
- Agent A 发送了消息
- Agent B 没有收到
- Agent A 不知道消息没送达
- 系统陷入等待

原因：
- 消息队列问题
- 网络问题
- 处理超时
"""

# 解决方案：确认机制
MESSAGE_CONFIRMATION = """
实现：
1. 发送消息时设置超时
2. 等待确认回复
3. 超时后重试或告警

示例代码：
async def send_with_confirmation(message, recipient, timeout=30):
    message_id = await send_message(message, recipient)

    # 等待确认
    confirmation = await wait_for_confirmation(
        message_id,
        timeout=timeout
    )

    if not confirmation:
        # 重试或告警
        await retry_or_alert(message_id)

    return confirmation
"""


# 错误2：消息格式不一致

MESSAGE_FORMAT_INCONSISTENCY = """
Agent A 发送：
{
    "type": "task_result",
    "data": "success",
    "result": {...}
}

Agent B 期望：
{
    "type": "result",
    "status": "completed",
    "output": {...}
}

问题：字段名不一致，解析失败
"""

# 解决方案：标准化协议
MESSAGE_PROTOCOL = """
定义统一的消息格式：

# 请求消息
{
    "msg_id": "uuid",
    "type": "request",
    "action": "task_name",
    "data": {...},
    "timestamp": "ISO8601"
}

# 响应消息
{
    "msg_id": "uuid",  # 对应请求的 msg_id
    "type": "response",
    "status": "success|error",
    "result": {...},
    "error": {...},  # 如果失败
    "timestamp": "ISO8601"
}

# 确认消息
{
    "msg_id": "uuid",  # 对应请求的 msg_id
    "type": "ack",
    "timestamp": "ISO8601"
}
"""
```

### 12.3 安全错误

```python
# 错误1：生成不安全代码

INSECURE_CODE_GENERATION = """
# AI 生成的代码（不安全）

def get_user(user_id):
    query = f"SELECT * FROM users WHERE id = {user_id}"
    return db.execute(query)

# 问题：SQL 注入漏洞
"""

# 正确做法：添加安全检查
SECURE_CODE_GENERATION = """
# Claude Code 应该：
1. 检测到潜在的安全问题
2. 标记为安全问题
3. 提供安全的替代方案

安全版本：
def get_user(user_id):
    query = "SELECT * FROM users WHERE id = %s"
    return db.execute(query, (user_id,))
"""


# 错误2：敏感信息泄露

SECRET_EXPOSURE = """
# Claude Code 不应该生成：

API_KEY = "sk-xxxxxxxxxxxxxxxxxxxx"
PASSWORD = "admin123"
DATABASE_URL = "postgresql://user:password@host/db"

问题：
- 密钥暴露在代码中
- 可能被提交到版本控制
- 造成安全风险
"""

# 正确做法
SECRET_MANAGEMENT = """
# Claude Code 应该生成：

import os

API_KEY = os.environ.get("API_KEY")
if not API_KEY:
    raise ValueError("API_KEY environment variable not set")

# 或者使用专门的密钥管理
from secret_manager import get_secret
API_KEY = get_secret("api_key")
"""


# 错误3：权限过度开放

EXCESSIVE_PERMISSIONS = """
# Claude Code 被赋予过多权限

permissions = {
    "allow_delete": True,  # 可以删除任何文件
    "allow_exec": True,   # 可以执行任何命令
    "allow_network": True  # 可以访问任何网络
}

风险：
- 如果被恶意利用，后果严重
- 误操作可能导致数据丢失
"""

# 正确做法：最小权限原则
MINIMAL_PERMISSIONS = """
permissions = {
    "allow_delete": False,  # 禁止删除
    "allow_exec": False,    # 禁止执行命令
    "allow_network": True,  # 仅允许必要的网络访问
    "readonly_paths": ["./src", "./tests"],
    "allowed_commands": ["git", "npm", "pytest"]
}
"""
```

### 12.5 错误恢复策略

```python
# 策略1：检查点机制

CHECKPOINT_STRATEGY = """
问题：长时间任务失败后需要从头开始

解决方案：检查点机制

实现：
class CheckpointableTask:
    def __init__(self, checkpoint_file):
        self.checkpoint_file = checkpoint_file

    async def save_checkpoint(self, state):
        with open(self.checkpoint_file, 'w') as f:
            json.dump(state, f)

    async def load_checkpoint(self):
        if os.path.exists(self.checkpoint_file):
            with open(self.checkpoint_file, 'r') as f:
                return json.load(f)
        return None

    async def execute(self):
        # 尝试加载检查点
        state = await self.load_checkpoint()

        if state:
            print(f"从检查点恢复：步骤 {state['step']}")
            start_step = state['step'] + 1
        else:
            state = {"step": 0}
            start_step = 0

        for step in range(start_step, total_steps):
            # 执行步骤
            await execute_step(step)

            # 保存检查点
            state["step"] = step
            await self.save_checkpoint(state)

        return state
"""


# 策略2：重试机制

RETRY_STRATEGY = """
问题：网络请求可能失败

解决方案：指数退避重试

实现：
async def retry_with_backoff(
    func,
    max_retries=3,
    base_delay=1,
    max_delay=60
):
    for attempt in range(max_retries):
        try:
            return await func()
        except RetryableError as e:
            if attempt == max_retries - 1:
                raise

            delay = min(base_delay * (2 ** attempt), max_delay)
            print(f"尝试 {attempt + 1} 失败，{delay}s 后重试...")
            await asyncio.sleep(delay)


# 使用示例
async def fetch_data(url):
    async def _fetch():
        async with aiohttp.ClientSession() as session:
            async with session.get(url) as response:
                return await response.json()

    return await retry_with_backoff(_fetch)
"""


# 策略3：回退机制

FALLBACK_STRATEGY = """
问题：高级功能失败

解决方案：回退到简单实现

实现：
async def process_with_fallback(data):
    # 尝试使用高级方法
    try:
        return await process_advanced(data)
    except AdvancedMethodError as e:
        logger.warning(f"高级方法失败，{e}，使用简单方法")

        # 回退到简单方法
        return await process_simple(data)


# 示例：AI 模型回退
async def generate_with_fallback(prompt):
    # 尝试使用最强的模型
    try:
        return await call_model("gpt-4", prompt)
    except ModelError as e:
        logger.warning(f"GPT-4 不可用，{e}，回退到 GPT-3.5")

        try:
            return await call_model("gpt-3.5-turbo", prompt)
        except ModelError:
            logger.error("所有模型都不可用")
            raise
"""
```
