### 20.1 配置管理概述

#### 为什么需要配置管理？

```
问题场景：

1. 环境差异
   - 开发环境
   - 测试环境
   - 生产环境
   不同的配置容易出错

2. 配置泄露
   - API Key 硬编码
   - 密码在代码中
   - 敏感信息进 Git

3. 配置混乱
   - 多个地方都有配置
   - 不知道哪个生效
   - 难以追踪变更
```

#### 配置管理原则

```
最佳实践：

1. 与环境分离
   - 配置不写在代码中
   - 通过环境变量注入
   - 使用配置中心

2. 最小权限
   - 只给必要的权限
   - 定期审查权限
   - 及时撤销

3. 可追溯
   - 配置变更有记录
   - 可以回滚
   - 审计日志
```

### 20.2 多环境配置

```python
class ConfigManager:
    """配置管理器"""

    def __init__(self, environment: str = "development"):
        self.environment = environment
        self.config = {}
        self.load_config()

    def load_config(self):
        """加载配置"""
        # 1. 加载默认配置
        default_config = self._load_file("config/default.json")
        self.config.update(default_config)

        # 2. 加载环境配置
        env_config = self._load_file(f"config/{self.environment}.json")
        self.config.update(env_config)

        # 3. 加载本地覆盖
        local_config = self._load_file("config/local.json")
        if local_config:
            self.config.update(local_config)

        # 4. 环境变量覆盖
        self._apply_env_overrides()

    def _apply_env_overrides(self):
        """应用环境变量覆盖"""
        # 数据库配置
        if os.environ.get("DATABASE_URL"):
            self.config["database"]["url"] = os.environ["DATABASE_URL"]

        # API Key
        if os.environ.get("API_KEY"):
            self.config["api"]["key"] = os.environ["API_KEY"]

        # 调试模式
        if os.environ.get("DEBUG"):
            self.config["debug"] = os.environ["DEBUG"].lower() == "true"

    def get(self, key: str, default: Any = None) -> Any:
        """获取配置"""
        keys = key.split(".")
        value = self.config

        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
                if value is None:
                    return default
            else:
                return default

        return value

    def _load_file(self, path: str) -> dict:
        """从文件加载配置"""
        if not os.path.exists(path):
            return {}

        with open(path, "r") as f:
            return json.load(f)
```

### 20.3 敏感信息管理

```python
class SecretManager:
    """敏感信息管理器"""

    def __init__(self):
        self.cache = {}
        self.encryption_key = self._get_encryption_key()

    def get_secret(self, key: str) -> str:
        """获取敏感信息"""
        if key in self.cache:
            return self.cache[key]

        # 从环境变量获取
        value = os.environ.get(key.upper())

        if not value:
            # 从密钥管理器获取（如 Vault, AWS Secrets Manager）
            value = self._fetch_from_vault(key)

        if not value:
            raise SecretNotFoundError(f"Secret not found: {key}")

        # 缓存
        self.cache[key] = value
        return value

    def _get_encryption_key(self) -> bytes:
        """获取加密密钥"""
        key = os.environ.get("ENCRYPTION_KEY")
        if key:
            return base64.b64decode(key)
        return os.urandom(32)

    def encrypt(self, value: str) -> str:
        """加密敏感信息"""
        from cryptography.fernet import Fernet

        f = Fernet(self.encryption_key)
        return f.encrypt(value.encode()).decode()

    def decrypt(self, encrypted: str) -> str:
        """解密敏感信息"""
        from cryptography.fernet import Fernet

        f = Fernet(self.encryption_key)
        return f.decrypt(encrypted.encode()).decode()

class AWSSecretsManager:
    """AWS 密钥管理器"""

    def __init__(self):
        self.client = boto3.client("secretsmanager")

    def get_secret(self, name: str) -> str:
        """从 AWS Secrets Manager 获取密钥"""
        response = self.client.get_secret_value(SecretId=name)
        return response["SecretString"]

    def create_secret(self, name: str, value: str):
        """创建密钥"""
        self.client.create_secret(
            Name=name,
            SecretString=value
        )

    def rotate_secret(self, name: str):
        """轮换密钥"""
        self.client.rotate_secret(SecretId=name)
```

```python
class Permission:
    """权限定义"""

    READ = "read"
    WRITE = "write"
    DELETE = "delete"
    EXECUTE = "execute"
    ADMIN = "admin"


class RBAC:
    """基于角色的访问控制"""

    def __init__(self):
        self.roles = {}
        self.permissions = {}
        self.user_roles = {}

    def define_role(self, role_name: str, permissions: List[str]):
        """定义角色"""
        self.roles[role_name] = permissions

    def assign_role(self, user: str, role_name: str):
        """分配角色"""
        if user not in self.user_roles:
            self.user_roles[user] = []
        self.user_roles[user].append(role_name)

    def has_permission(self, user: str, permission: str) -> bool:
        """检查权限"""
        if user not in self.user_roles:
            return False

        for role_name in self.user_roles[user]:
            if role_name in self.roles:
                if permission in self.roles[role_name]:
                    return True

        return False

    def check_or_raise(self, user: str, permission: str):
        """检查权限，否则抛出异常"""
        if not self.has_permission(user, permission):
            raise PermissionDeniedError(
                f"User {user} does not have {permission} permission"
            )
```

### 20.5 审计日志

```python
import hashlib
from datetime import datetime


class AuditLogger:
    """审计日志"""

    def __init__(self, storage: AuditStorage):
        self.storage = storage

    def log(
        self,
        event_type: str,
        user: str,
        resource: str,
        action: str,
        result: str,
        details: dict = None
    ):
        """记录审计日志"""
        entry = AuditEntry(
            timestamp=datetime.now().isoformat(),
            event_type=event_type,
            user=user,
            resource=resource,
            action=action,
            result=result,
            details=details or {},
            ip_address=self._get_client_ip(),
            user_agent=self._get_user_agent()
        )

        # 计算哈希保证完整性
        entry.hash = self._calculate_hash(entry)

        self.storage.save(entry)

    def _calculate_hash(self, entry: AuditEntry) -> str:
        """计算日志条目哈希"""
        content = (
            f"{entry.timestamp}"
            f"{entry.event_type}"
            f"{entry.user}"
            f"{entry.resource}"
            f"{entry.action}"
        )
        return hashlib.sha256(content.encode()).hexdigest()

    def query(
        self,
        user: str = None,
        event_type: str = None,
        start_time: datetime = None,
        end_time: datetime = None
    ) -> List[AuditEntry]:
        """查询审计日志"""
        return self.storage.query(
            user=user,
            event_type=event_type,
            start_time=start_time,
            end_time=end_time
        )

    def verify_integrity(self) -> bool:
        """验证日志完整性"""
        entries = self.storage.get_all()

        for entry in entries:
            expected_hash = self._calculate_hash(entry)
            if entry.hash != expected_hash:
                return False

        return True


@dataclass
class AuditEntry:
    """审计条目"""
    timestamp: str
    event_type: str
    user: str
    resource: str
    action: str
    result: str
    details: dict
    ip_address: str
    user_agent: str
    hash: str = None
```

```python
class SecurityScanner:
    """安全扫描器"""

    def __init__(self):
        self.rules = SecurityRules()

    async def scan_file(self, file_path: str) -> List[SecurityIssue]:
        """扫描单个文件"""
        issues = []

        content = Path(file_path).read_text()

        # 检查硬编码密钥
        issues.extend(self._check_hardcoded_secrets(content, file_path))

        # 检查 SQL 注入风险
        issues.extend(self._check_sql_injection(content, file_path))

        # 检查 XSS 风险
        issues.extend(self._check_xss_risks(content, file_path))

        # 检查敏感数据暴露
        issues.extend(self._check_sensitive_data(content, file_path))

        return issues

    def _check_hardcoded_secrets(
        self,
        content: str,
        file_path: str
    ) -> List[SecurityIssue]:
        """检查硬编码密钥"""
        issues = []

        patterns = [
            (r'api[_-]?key\s*=\s*["\'][a-zA-Z0-9]{20,}["\']', "API Key"),
            (r'password\s*=\s*["\'][^"\']{8,}["\']', "Password"),
            (r'secret\s*=\s*["\'][a-zA-Z0-9]{20,}["\']', "Secret"),
            (r'token\s*=\s*["\'][a-zA-Z0-9]{20,}["\']', "Token"),
        ]

        for pattern, secret_type in patterns:
            for match in re.finditer(pattern, content, re.IGNORECASE):
                line_no = content[:match.start()].count("\n") + 1
                issues.append(SecurityIssue(
                    severity="critical",
                    type=f"hardcoded_{secret_type.lower()}",
                    file=file_path,
                    line=line_no,
                    message=f"发现硬编码 {secret_type}",
                    remediation=f"使用环境变量或密钥管理器"
                ))

        return issues

    def _check_sql_injection(
        self,
        content: str,
        file_path: str
    ) -> List[SecurityIssue]:
        """检查 SQL 注入"""
        issues = []

        # 检查 f-string 拼接 SQL
        sql_patterns = [
            r'f["\']SELECT.*\{.*\}',
            r'f["\'].*INSERT.*\{.*\}',
            r'f["\'].*UPDATE.*\{.*\}',
            r'f["\'].*DELETE.*\{.*\}',
        ]

        for pattern in sql_patterns:
            for match in re.finditer(pattern, content, re.IGNORECASE):
                line_no = content[:match.start()].count("\n") + 1
                issues.append(SecurityIssue(
                    severity="high",
                    type="sql_injection",
                    file=file_path,
                    line=line_no,
                    message="潜在的 SQL 注入风险",
                    remediation="使用参数化查询"
                ))

        return issues


@dataclass
class SecurityIssue:
    """安全问题"""
    severity: str  # critical, high, medium, low
    type: str
    file: str
    line: int
    message: str
    remediation: str
```

```python
        report = ComplianceReport(framework=framework)

            status = self._check_requirement(requirement)
            report.add_requirement(requirement, status)

        return report

    def _check_requirement(self, requirement: str) -> ComplianceStatus:
        """检查单项要求"""

            status="pass",  # pass, fail, partial

class ComplianceReport:
    """合规报告"""
    framework: str
    requirements: List[Dict] = field(default_factory=list)

    def add_requirement(self, name: str, status: ComplianceStatus):
        self.requirements.append({
            "name": name,
            "status": status.status,
            "evidence": status.evidence
        })

    def get_pass_rate(self) -> float:
        """获取通过率"""
        if not self.requirements:
            return 0
        passed = sum(1 for r in self.requirements if r["status"] == "pass")
        return passed / len(self.requirements)
```
