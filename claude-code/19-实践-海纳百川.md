### 17.1 插件系统概述

#### 什么是 Claude Code 插件？

```
Claude Code 插件 = 扩展 Claude Code 功能的模块

插件类型：
1. 内置插件（Built-in）
   - Claude Code 自带
   - 无需安装，直接使用

2. 第三方插件（Third-party）
   - 社区开发
   - 需要安装

3. 自定义插件（Custom）
   - 自己开发
   - 仅限本地使用
```

#### 插件 vs Skills

```
对比：

Skills：
- 轻量级
- 配置文件（YAML/Markdown）
- 通过指令触发
- 适合简单任务自动化

插件：
- 功能完整
- 代码实现
- 深度集成
- 适合复杂功能扩展
```

### 17.2 插件架构

```python
class Plugin:
    """插件基类"""

    def __init__(self, config: dict = None):
        self.config = config or {}
        self.enabled = True

    def on_install(self):
        """插件安装时调用"""
        pass

    def on_uninstall(self):
        """插件卸载时调用"""
        pass

    def on_enable(self):
        """插件启用时调用"""
        pass

    def on_disable(self):
        """插件禁用时调用"""
        pass

    def on_message(self, message: str) -> str:
        """处理消息"""
        return message


class PluginManager:
    """插件管理器"""

    def __init__(self):
        self.plugins: Dict[str, Plugin] = {}
        self.hooks: Dict[str, List[callable]] = {}

    def register_plugin(self, plugin_id: str, plugin: Plugin):
        """注册插件"""
        self.plugins[plugin_id] = plugin
        plugin.on_install()

    def unregister_plugin(self, plugin_id: str):
        """卸载插件"""
        if plugin_id in self.plugins:
            self.plugins[plugin_id].on_uninstall()
            del self.plugins[plugin_id]

    def enable_plugin(self, plugin_id: str):
        """启用插件"""
        if plugin_id in self.plugins:
            self.plugins[plugin_id].on_enable()

    def disable_plugin(self, plugin_id: str):
        """禁用插件"""
        if plugin_id in self.plugins:
            self.plugins[plugin_id].on_disable()

    def register_hook(self, hook_name: str, callback: callable):
        """注册钩子"""
        if hook_name not in self.hooks:
            self.hooks[hook_name] = []
        self.hooks[hook_name].append(callback)

    def trigger_hook(self, hook_name: str, *args, **kwargs):
        """触发钩子"""
        results = []
        for callback in self.hooks.get(hook_name, []):
            result = callback(*args, **kwargs)
            results.append(result)
        return results
```

# main.py
from plugin import Plugin

class MyPlugin(Plugin):
    """我的插件"""

    def __init__(self, config):
        super().__init__(config)
        self.name = "MyPlugin"

    def on_install(self):
        """安装时初始化"""
        print(f"{self.name} installed")

    def on_message(self, message: str) -> str:
        """处理消息"""
        if message.startswith("!hello"):
            return "Hello from plugin!"
        return message

    def on_command(self, command: str, args: list) -> str:
        """处理命令"""
        if command == "greet":
            name = args[0] if args else "World"
            return f"Hello, {name}!"
        return None

### 17.4 插件生命周期

```python
class PluginLifecycle:
    """插件生命周期管理"""

    STATES = {
        "installed": "已安装",
        "enabled": "已启用",
        "disabled": "已禁用",
        "running": "运行中",
        "error": "错误"
    }

    def __init__(self, plugin: Plugin):
        self.plugin = plugin
        self.state = "installed"
        self.error_log = []

    def enable(self):
        """启用插件"""
        if self.state == "disabled":
            self.plugin.on_enable()
            self.state = "enabled"
        elif self.state == "installed":
            self.state = "enabled"
            self.plugin.on_enable()

    def disable(self):
        """禁用插件"""
        if self.state in ("enabled", "running"):
            self.plugin.on_disable()
            self.state = "disabled"

    def start(self):
        """启动插件"""
        if self.state == "enabled":
            self.state = "running"

    def stop(self):
        """停止插件"""
        if self.state == "running":
            self.state = "enabled"

    def uninstall(self):
        """卸载插件"""
        self.disable()
        self.plugin.on_uninstall()
        self.state = "uninstalled"
```

### 17.5 插件通信

```python
class PluginAPI:
    """插件 API"""

    def __init__(self, plugin_id: str):
        self.plugin_id = plugin_id

    def send_message(self, target: str, message: str):
        """发送消息到其他插件"""
        plugin_manager.send_to_plugin(target, {
            "from": self.plugin_id,
            "type": "message",
            "content": message
        })

    def emit_event(self, event: str, data: dict = None):
        """发送事件"""
        plugin_manager.trigger_hook(f"event_{event}", {
            "from": self.plugin_id,
            "data": data
        })

    def store_data(self, key: str, value: any):
        """存储数据"""
        storage.set(f"{self.plugin_id}:{key}", value)

    def get_data(self, key: str, default: any = None):
        """获取数据"""
        return storage.get(f"{self.plugin_id}:{key}", default)

    def get_config(self, key: str, default: any = None):
        """获取配置"""
        return self.plugin.config.get(key, default)
```

class PluginMarketplace:
    """插件市场"""

    def __init__(self):
        self.api_base = "https://api.claude-code/plugins"

    async def search(self, query: str) -> List[PluginInfo]:
        """搜索插件"""
        response = await http.get(
            f"{self.api_base}/search",
            params={"q": query}
        )
        return response.json()

    async def install(self, plugin_id: str):
        """安装插件"""
        # 1. 获取插件信息
        info = await self.get_plugin_info(plugin_id)

        # 2. 下载插件包
        await self.download_plugin(plugin_id, info.download_url)

        # 3. 验证插件
        if not self.verify_plugin(plugin_id):
            raise VerificationError("Plugin verification failed")

        # 4. 安装插件
        plugin_manager.register_plugin(plugin_id, info.entry_class)

    async def uninstall(self, plugin_id: str):
        """卸载插件"""
        plugin_manager.unregister_plugin(plugin_id)
        storage.cleanup_plugin_data(plugin_id)

    async def update(self, plugin_id: str):
        """更新插件"""
        current = plugin_manager.plugins[plugin_id]
        latest = await self.get_plugin_info(plugin_id)

        if latest.version > current.version:
            await self.uninstall(plugin_id)
            await self.install(plugin_id)

### 17.8 安全机制

```python
class PluginSecurity:
    """插件安全"""

    @staticmethod
    def validate_permissions(plugin_id: str, required: List[str]) -> bool:
        """验证权限"""
        manifest = PluginManifest.load(plugin_id)
        return all(p in manifest.permissions for p in required)

    @staticmethod
    def sandbox_plugin(plugin: Plugin) -> Plugin:
        """沙箱化插件"""
        # 限制文件访问
        original_read = plugin.read_file
        original_write = plugin.write_file

        def safe_read(path):
            if not SecurityPolicy.allow_file(path, 'read'):
                raise PermissionError(f"Read not allowed: {path}")
            return original_read(path)

        def safe_write(path, content):
            if not SecurityPolicy.allow_file(path, 'write'):
                raise PermissionError(f"Write not allowed: {path}")
            return original_write(path, content)

        plugin.read_file = safe_read
        plugin.write_file = safe_write

        return plugin


class SecurityPolicy:
    """安全策略"""

    ALLOWED_PATHS = [
        "./src",
        "./tests",
        "./docs"
    ]

    DENIED_PATTERNS = [
        "*.key",
        "*.pem",
        "*password*",
        "*secret*"
    ]

    @classmethod
    def allow_file(cls, path: str, operation: str) -> bool:
        """检查文件操作是否允许"""
        # 检查是否在允许路径
        for allowed in cls.ALLOWED_PATHS:
            if str(path).startswith(allowed):
                return True

        # 检查是否匹配拒绝模式
        for pattern in cls.DENIED_PATTERNS:
            if cls.match_pattern(path, pattern):
                return False

        return False

    @staticmethod
    def match_pattern(path: str, pattern: str) -> bool:
        """简单的模式匹配"""
        import fnmatch
        return fnmatch.fnmatch(path.lower(), pattern.lower())
```
