### 14.1 调用链路基础

#### 什么是调用链路？

```
调用链路 = 函数/方法之间的调用关系链

示例：
main() 
  └─> init()
       ├─> load_config()
       └─> setup_database()
            └─> connect()

调用链路：[main -> init -> setup_database -> connect]
```

#### 为什么需要追踪调用链路？

```
典型应用场景：

1. Bug 定位
   - 错误在某处发生，但不知道从哪调用来的
   - 需要追踪完整调用栈

2. 性能分析
   - 知道某函数慢，但不知道被谁调用最频繁
   - 需要分析调用链路的性能分布

3. 代码理解
   - 新项目，不知道代码执行流程
   - 需要理解模块间的调用关系

4. 安全审计
   - 检查敏感函数的调用路径
   - 确保没有未授权的调用链
```

### 14.2 调用图数据结构

```python
from dataclasses import dataclass, field
from typing import List, Set, Dict, Optional, Any
from collections import defaultdict
import networkx as nx


@dataclass
class CallNode:
    """调用图节点"""
    name: str
    type: str  # "function", "method", "class"
    file_path: str
    line_number: int
    signature: str  # 函数签名
    docstring: Optional[str] = None


@dataclass
class CallEdge:
    """调用边"""
    caller: str  # 调用者
    callee: str  # 被调用者
    call_site: str  # 调用位置（文件:行号）
    is_conditional: bool = False  # 是否条件调用
    is_async: bool = False  # 是否异步调用


class CallGraph:
    """调用图"""

    def __init__(self):
        self.nodes: Dict[str, CallNode] = {}
        self.edges: List[CallEdge] = []
        self.graph: nx.DiGraph = nx.DiGraph()

    def add_node(self, node: CallNode):
        """添加节点"""
        self.nodes[node.name] = node
        self.graph.add_node(node.name, **{
            'file': node.file_path,
            'line': node.line_number,
            'type': node.type
        })

    def add_edge(self, edge: CallEdge):
        """添加边"""
        self.edges.append(edge)
        self.graph.add_edge(edge.caller, edge.callee)

    def get_callers(self, node_name: str) -> List[str]:
        """获取调用该节点的所有节点"""
        return list(self.graph.predecessors(node_name))

    def get_callees(self, node_name: str) -> List[str]:
        """获取该节点调用的所有节点"""
        return list(self.graph.successors(node_name))

    def find_path(self, start: str, end: str) -> List[str]:
        """查找从 start 到 end 的路径"""
        try:
            return list(nx.shortest_path(self.graph, start, end))
        except nx.NetworkXNoPath:
            return []

    def find_all_paths(self, start: str, end: str) -> List[List[str]]:
        """查找所有从 start 到 end 的路径"""
        try:
            return [list(p) for p in nx.all_simple_paths(self.graph, start, end)]
        except nx.NetworkXNoPath:
            return []

    def get_call_chain(self, start: str, depth: int = 10) -> Dict[str, Any]:
        """获取调用链"""
        if start not in self.graph:
            return {}

        # BFS 遍历
        visited = {start: 0}
        queue = [start]
        levels = defaultdict(list)

        while queue:
            node = queue.pop(0)
            level = visited[node]

            if level > depth:
                continue

            levels[level].append(node)

            for neighbor in self.graph.successors(node):
                if neighbor not in visited:
                    visited[neighbor] = level + 1
                    queue.append(neighbor)

        return dict(levels)
```

### 14.3 静态调用分析

#### 原理

```
静态分析 = 不运行代码，通过分析源代码找出调用关系

优点：
- 不需要运行代码
- 可以分析所有可能的执行路径
- 可以分析完整代码库

缺点：
- 无法处理动态派发（反射、虚函数）
- 无法处理运行时才确定的调用
- 可能产生误报（死代码）
```

#### Python 静态分析实现

```python
class PythonCallGraphAnalyzer:
    """Python 静态调用图分析器"""

    def __init__(self):
        self.graph = CallGraph()
        self.current_file = None

    def analyze(self, file_path: Path) -> CallGraph:
        """分析 Python 文件"""
        self.current_file = file_path
        content = file_path.read_text()

        # 解析 AST
        tree = ast.parse(content)

        # 分析函数定义
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                self._analyze_function(node, tree)

        return self.graph

    def _analyze_function(self, func_node: ast.FunctionDef, tree: ast.AST):
        """分析函数定义"""
        func_name = func_node.name
        file_path = str(self.current_file)

        # 获取函数签名
        args = [arg.arg for arg in func_node.args.args]
        signature = f"{func_name}({', '.join(args)})"

        # 创建节点
        node = CallNode(
            name=func_name,
            type="function",
            file_path=file_path,
            line_number=func_node.lineno,
            signature=signature,
            docstring=ast.get_docstring(func_node)
        )

        self.graph.add_node(node)

        # 分析函数体中的调用
        for child in ast.walk(func_node):
            if isinstance(child, ast.Call):
                self._analyze_call(func_node.name, child)

    def _analyze_call(self, caller: str, call_node: ast.Call):
        """分析函数调用"""
        callee = self._get_call_target(call_node)

        if callee:
            edge = CallEdge(
                caller=caller,
                callee=callee,
                call_site=f"{self.current_file}:{call_node.lineno}",
                is_async=isinstance(call_node, ast.Await) if hasattr(ast, 'Await') else False
            )
            self.graph.add_edge(edge)

    def _get_call_target(self, call_node: ast.Call) -> Optional[str]:
        """获取调用目标"""
        if isinstance(call_node.func, ast.Name):
            # 直接调用：func()
            return call_node.func.id
        elif isinstance(call_node.func, ast.Attribute):
            # 方法调用：obj.method()
            return call_node.func.attr
        return None

    def analyze_project(self, project_path: Path) -> CallGraph:
        """分析整个项目"""
        # 收集所有 Python 文件
        py_files = list(project_path.rglob("*.py"))

        for py_file in py_files:
            # 跳过测试文件和虚拟环境
            if self._should_skip(py_file):
                continue

            try:
                self.analyze(py_file)
            except Exception as e:
                print(f"Error analyzing {py_file}: {e}")

        # 解析 import 语句，建立跨文件调用关系
        self._resolve_imports(project_path)

        return self.graph

    def _should_skip(self, file_path: Path) -> bool:
        """判断是否应该跳过"""
        skip_patterns = [
            '__pycache__',
            '.venv',
            'venv',
            'env',
            'tests',
            'test_',
            '_test.py'
        ]

        path_str = str(file_path)
        return any(pattern in path_str for pattern in skip_patterns)

    def _resolve_imports(self, project_path: Path):
        """解析跨文件调用"""
        # 读取所有文件的导入
        imports_map = self._collect_imports(project_path)

        # 更新调用图中的外部引用
```

#### JavaScript/TypeScript 静态分析

```python
class JavaScriptCallGraphAnalyzer:
    """JavaScript 调用图分析器"""

    def __init__(self):
        self.graph = CallGraph()

    def analyze(self, file_path: Path) -> CallGraph:
        """分析 JavaScript/TypeScript 文件"""
        content = file_path.read_text()

        # 使用正则表达式提取函数定义和调用
        self._analyze_function_definitions(content, file_path)
        self._analyze_function_calls(content, file_path)

        return self.graph

    def _analyze_function_definitions(self, content: str, file_path: Path):
        """分析函数定义"""
        # 函数声明：function name() {}
        func_pattern = r'function\s+(\w+)\s*\([^)]*\)\s*\{'

        for match in re.finditer(func_pattern, content):
            func_name = match.group(1)
            line_number = content[:match.start()].count('\n') + 1

            node = CallNode(
                name=func_name,
                type="function",
                file_path=str(file_path),
                line_number=line_number,
                signature=match.group(0)[:50]
            )
            self.graph.add_node(node)

        # 箭头函数：const name = () => {}
        arrow_pattern = r'const\s+(\w+)\s*=\s*\([^)]*\)\s*=>'

        for match in re.finditer(arrow_pattern, content):
            func_name = match.group(1)
            line_number = content[:match.start()].count('\n') + 1

            node = CallNode(
                name=func_name,
                type="arrow_function",
                file_path=str(file_path),
                line_number=line_number,
                signature=match.group(0)[:50]
            )
            self.graph.add_node(node)

    def _analyze_function_calls(self, content: str, file_path: Path):
        """分析函数调用"""
        # 函数调用：name()
        call_pattern = r'(\w+)\s*\([^)]*\)'

        for match in re.finditer(call_pattern, content):
            caller_name = match.group(1)
            line_number = content[:match.start()].count('\n') + 1

            # 检查是否是已知函数
            if caller_name in self.graph.nodes:
                edge = CallEdge(
                    caller=caller_name,
                    callee=match.group(1),
                    call_site=f"{file_path}:{line_number}"
                )
                self.graph.add_edge(edge)
```

### 14.4 动态调用追踪

#### 原理

```
动态追踪 = 在程序运行时记录实际的调用关系

优点：
- 精确，只记录真实发生的调用
- 可以处理反射和动态派发
- 可以获取运行时信息（参数、返回值）

缺点：
- 需要运行代码
- 可能遗漏未执行的代码路径
- 性能开销
```

#### Python 装饰器追踪

```python
import functools
import time
import traceback
from typing import Any, Callable, Dict, List
from dataclasses import dataclass, field
from datetime import datetime


@dataclass
class CallRecord:
    """调用记录"""
    function_name: str
    args: tuple
    kwargs: Dict[str, Any]
    return_value: Any = None
    exception: Exception = None
    start_time: float = field(default_factory=time.time)
    end_time: float = None
    duration: float = None
    call_stack: List[str] = field(default_factory=list)


class CallTracer:
    """函数调用追踪器"""

    def __init__(self, max_depth: int = 100):
        self.max_depth = max_depth
        self.records: List[CallRecord] = []
        self.call_stack: List[str] = []
        self.enabled = False

    def trace(self, func: Callable) -> Callable:
        """装饰器：追踪函数调用"""

        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if not self.enabled:
                return func(*args, **kwargs)

            # 记录调用栈
            self.call_stack.append(func.__name__)

            # 限制栈深度
            if len(self.call_stack) > self.max_depth:
                self.call_stack.pop()
                return func(*args, **kwargs)

            record = CallRecord(
                function_name=func.__name__,
                args=args,
                kwargs=kwargs,
                call_stack=traceback.format_stack()[:-1]  # 排除当前栈
            )

            try:
                result = func(*args, **kwargs)
                record.return_value = result
                return result

            except Exception as e:
                record.exception = e
                raise

            finally:
                record.end_time = time.time()
                record.duration = record.end_time - record.start_time
                self.records.append(record)
                self.call_stack.pop()

        return wrapper

    def enable(self):
        """启用追踪"""
        self.enabled = True

    def disable(self):
        """禁用追踪"""
        self.enabled = False

    def get_report(self) -> str:
        """生成追踪报告"""
        if not self.records:
            return "No calls recorded"

        report = ["# 调用追踪报告\n"]

        # 按耗时排序
        sorted_records = sorted(
            self.records,
            key=lambda r: r.duration or 0,
            reverse=True
        )

        report.append(f"总调用次数：{len(self.records)}")
        report.append(f"总耗时：{sum(r.duration or 0 for r in self.records):.3f}s\n")

        report.append("## 最耗时调用\n")
        for i, record in enumerate(sorted_records[:10], 1):
            report.append(f"{i}. {record.function_name}")
            report.append(f"   耗时：{record.duration:.3f}s")
            if record.exception:
                report.append(f"   异常：{record.exception}")
            report.append("")

        return "\n".join(report)


# 全局追踪器
_global_tracer = CallTracer()


def trace(func: Callable) -> Callable:
    """便捷装饰器"""
    return _global_tracer.trace(func)


# 使用示例
tracer = CallTracer()
tracer.enable()

@tracer.trace
def slow_function():
    time.sleep(0.1)
    return "done"

@tracer.trace
def caller():
    return slow_function()

caller()
print(tracer.get_report())
```

#### 异步调用追踪

```python
import asyncio
from contextlib import asynccontextmanager


class AsyncCallTracer:
    """异步调用追踪器"""

    def __init__(self):
        self.records: List[Dict] = []
        self.task_stack: List[str] = []

    @asynccontextmanager
    async def trace(self, name: str):
        """追踪异步函数"""
        self.task_stack.append(name)
        start = asyncio.get_event_loop().time()

        record = {
            'name': name,
            'start': start,
            'task': asyncio.current_task(),
            'stack': list(self.task_stack)
        }

        try:
            yield record
        finally:
            end = asyncio.get_event_loop().time()
            record['duration'] = end - start
            self.records.append(record)
            self.task_stack.pop()

    def get_correlation_chain(self, task_id: str = None) -> List[Dict]:
        """获取关联的调用链"""
        if task_id:
            return [r for r in self.records if r.get('task') == task_id]

        # 返回当前任务的调用链
        current = asyncio.current_task()
        return [r for r in self.records if r.get('task') == current]


# 使用示例
async def main():
    tracer = AsyncCallTracer()

    async def async_task(name: str):
        async with tracer.trace(name):
            await asyncio.sleep(0.1)

    async with tracer.trace('main'):
        await asyncio.gather(
            async_task('task1'),
            async_task('task2'),
            async_task('task3')
        )

    print(tracer.get_correlation_chain())
```

```python
def visualize_call_chain(call_records: List[CallRecord], max_depth: int = 10) -> str:
    """可视化调用链"""

    # 构建树结构
    root = {'name': 'root', 'children': []}
    stack = [root]

            'duration': record.duration,
            'children': []
        }

        if stack:
            stack[-1]['children'].append(node)

        stack.append(node)

        # 限制深度
        if len(stack) > max_depth:
            stack.pop(0)

    # 渲染文本树
    def render(node, prefix="", is_last=True):
        lines = []
        connector = "└── " if is_last else "├── "
        lines.append(f"{prefix}{connector}{node['name']} ({node.get('duration', 0):.3f}s)")

        children = node.get('children', [])
        for i, child in enumerate(children):
            new_prefix = prefix + ("    " if is_last else "│   ")
            lines.extend(render(child, new_prefix, i == len(children) - 1))

        return lines

    result = ["调用链：\n"]
    for i, child in enumerate(root['children']):
        result.extend(render(child, "", i == len(root['children']) - 1))

    return "\n".join(result)
```
