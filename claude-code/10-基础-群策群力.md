### 8.1 Agent Teams 的核心概念

#### 什么是 Agent Team？

```
Agent Team = 多个 Agent + 协作机制

核心要素：
1. 多个 Agent（至少 2 个）
2. 明确的角色分工
3. 通信协议
4. 协调机制
5. 共同目标
```

#### 为什么需要 Agent Teams？

```
单一 Agent 的局限：

1. 能力边界
   - 一个人不可能精通所有领域
   - 复杂任务需要多方面知识

2. 上下文限制
   - 所有任务共享同一个上下文
   - 信息过载导致性能下降

3. 效率问题
   - 所有任务串行执行
   - 无法利用并行优势

Agent Team 的优势：

1. 专业分工
   - 安全专家 Agent
   - 性能专家 Agent
   - 代码风格 Agent
   - 各司其职

2. 上下文隔离
   - 每个 Agent 有独立上下文
   - 避免信息过载

3. 并行协作
   - 多个 Agent 同时工作
   - 效率大幅提升
```

#### Agent Team vs Multi-Agent System

| 维度 | Agent Team | Multi-Agent System |
|------|-------------|-------------------|
| **目标** | 协作完成共同任务 | 各自独立目标 |
| **关系** | 队友关系 | 竞争/合作关系 |
| **通信** | 紧密协作 | 相对松散 |
| **场景** | 团队任务 | 模拟、博弈 |

from enum import Enum
from dataclasses import dataclass, field
from typing import Any, Optional, Dict, Set
import time
import uuid


class MessageType(Enum):
    """消息类型"""
    TASK = "task"                    # 任务分配
    RESULT = "result"                # 结果汇报
    QUERY = "query"                  # 查询请求
    RESPONSE = "response"            # 查询响应
    STATUS = "status"                # 状态报告
    ERROR = "error"                  # 错误报告
    APPROVAL = "approval"            # 审批请求
    APPROVED = "approved"            # 审批通过
    REJECTED = "rejected"           # 审批拒绝
    BROADCAST = "broadcast"          # 广播消息
    HEARTBEAT = "heartbeat"          # 心跳


class Priority(Enum):
    """消息优先级"""
    LOW = 0
    NORMAL = 1
    HIGH = 2
    CRITICAL = 3


@dataclass
class AgentMessage:
    """Agent 消息"""
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    type: MessageType = MessageType.MESSAGE
    from_agent: str = ""
    to_agent: Optional[str] = None  # None 表示广播
    content: Dict[str, Any] = field(default_factory=dict)
    priority: Priority = Priority.NORMAL
    timestamp: float = field(default_factory=time.time)
    correlation_id: Optional[str] = None  # 用于关联请求和响应
    ttl: int = 300  # 消息生存时间（秒）

    def is_expired(self) -> bool:
        """检查消息是否过期"""
        return time.time() - self.timestamp > self.ttl


@dataclass
class TaskMessage(AgentMessage):
    """任务消息"""
    def __init__(self, from_agent: str, to_agent: str, task_id: str,
                 task_data: Dict, priority: Priority = Priority.NORMAL):
        super().__init__(
            type=MessageType.TASK,
            from_agent=from_agent,
            to_agent=to_agent,
            content={"task_id": task_id, "task_data": task_data},
            priority=priority
        )


@dataclass
class ResultMessage(AgentMessage):
    """结果消息"""
    def __init__(self, from_agent: str, to_agent: str,
                 task_id: str, result: Any, success: bool = True):
        super().__init__(
            type=MessageType.RESULT,
            from_agent=from_agent,
            to_agent=to_agent,
            content={"task_id": task_id, "result": result, "success": success},
            priority=Priority.NORMAL
        )

#### 消息队列实现

```python
import asyncio
from typing import Dict, List, Optional, Callable
from collections import defaultdict
import logging

logger = logging.getLogger(__name__)


class MessageQueue:
    """Agent 消息队列"""

    def __init__(self):
        self.queues: Dict[str, asyncio.Queue] = defaultdict(asyncio.Queue)
        self.subscriptions: Dict[str, Set[str]] = defaultdict(set)
        self.handlers: Dict[str, Callable] = {}

    def subscribe(self, agent_id: str, channel: str):
        """订阅频道"""
        self.subscriptions[channel].add(agent_id)

    def unsubscribe(self, agent_id: str, channel: str):
        """取消订阅"""
        if agent_id in self.subscriptions[channel]:
            self.subscriptions[channel].remove(agent_id)

    async def send(self, message: AgentMessage):
        """发送消息"""
        if message.to_agent:
            # 点对点消息
            await self.queues[message.to_agent].put(message)
            logger.debug(f"Message {message.id} sent to {message.to_agent}")
        else:
            # 广播消息
            for agent_id in self.subscriptions.get(message.type.value, []):
                if agent_id != message.from_agent:
                    await self.queues[agent_id].put(message)
                    logger.debug(f"Broadcast {message.id} to {agent_id}")

    async def receive(self, agent_id: str, timeout: float = None) -> Optional[AgentMessage]:
        """接收消息"""
        try:
            if timeout:
                return await asyncio.wait_for(
                    self.queues[agent_id].get(),
                    timeout=timeout
                )
            else:
                return await self.queues[agent_id].get()
        except asyncio.TimeoutError:
            return None

    def register_handler(self, message_type: MessageType, handler: Callable):
        """注册消息处理器"""
        self.handlers[message_type.value] = handler

    async def process_message(self, agent_id: str):
        """处理消息"""
        message = await self.receive(agent_id)
        if message and message.type.value in self.handlers:
            await self.handlers[message.type.value](message)


class AgentMailbox:
    """Agent 邮箱"""

    def __init__(self, agent_id: str, queue: MessageQueue):
        self.agent_id = agent_id
        self.queue = queue
        self.inbox: List[AgentMessage] = []
        self.outbox: List[AgentMessage] = []

    async def send_to(self, to_agent: str, message_type: MessageType,
                      content: Dict, priority: Priority = Priority.NORMAL):
        """发送消息"""
        message = AgentMessage(
            type=message_type,
            from_agent=self.agent_id,
            to_agent=to_agent,
            content=content,
            priority=priority
        )
        self.outbox.append(message)
        await self.queue.send(message)

    async def broadcast(self, message_type: MessageType,
                       content: Dict, priority: Priority = Priority.NORMAL):
        """广播消息"""
        message = AgentMessage(
            type=message_type,
            from_agent=self.agent_id,
            to_agent=None,  # 广播
            content=content,
            priority=priority
        )
        self.outbox.append(message)
        await self.queue.send(message)

    async def receive(self, timeout: float = None) -> Optional[AgentMessage]:
        """接收消息"""
        message = await self.queue.receive(self.agent_id, timeout)
        if message:
            self.inbox.append(message)
        return message
```

### 8.4 团队管理器实现

```python
import asyncio
from typing import Dict, List, Optional, Set, Any
from dataclasses import dataclass, field
from enum import Enum
import logging

logger = logging.getLogger(__name__)


class AgentState(Enum):
    """Agent 状态"""
    IDLE = "idle"
    BUSY = "busy"
    WAITING = "waiting"
    OFFLINE = "offline"


@dataclass
class TeamMember:
    """团队成员"""
    agent_id: str
    name: str
    role: Role
    state: AgentState = AgentState.IDLE
    current_task: Optional[str] = None
    completed_tasks: List[str] = field(default_factory=list)
    mailbox: Optional[AgentMailbox] = None


class TeamManager:
    """团队管理器"""

    def __init__(self, team_name: str):
        self.team_name = team_name
        self.members: Dict[str, TeamMember] = {}
        self.team_lead: Optional[TeamMember] = None
        self.message_queue = MessageQueue()
        self.task_registry: Dict[str, Dict] = {}
        self._running = False

    def add_member(self, agent_id: str, name: str, role: Role) -> TeamMember:
        """添加团队成员"""
        mailbox = AgentMailbox(agent_id, self.message_queue)

        member = TeamMember(
            agent_id=agent_id,
            name=name,
            role=role,
            mailbox=mailbox
        )

        self.members[agent_id] = member

        # 如果是协调者角色，设置为主管
        if role.type == RoleType.COORDINATOR:
            self.team_lead = member

        logger.info(f"Added team member: {name} ({role.name})")

        return member

    def get_member(self, agent_id: str) -> Optional[TeamMember]:
        """获取团队成员"""
        return self.members.get(agent_id)

    def get_members_by_role(self, role_type: RoleType) -> List[TeamMember]:
        """按角色获取成员"""
        return [
            m for m in self.members.values()
            if m.role.type == role_type
        ]

    async def assign_task(self, task_id: str, assignee: str,
                        task_data: Dict, priority: Priority = Priority.NORMAL):
        """分配任务"""
        if assignee not in self.members:
            raise ValueError(f"Unknown agent: {assignee}")

        # 更新任务注册表
        self.task_registry[task_id] = {
            "task_id": task_id,
            "assignee": assignee,
            "status": "pending",
            "data": task_data,
            "priority": priority
        }

        # 更新成员状态
        member = self.members[assignee]
        member.state = AgentState.BUSY
        member.current_task = task_id

        # 发送任务消息
        await member.mailbox.send_to(
            assignee,
            MessageType.TASK,
            {
                "task_id": task_id,
                "task_data": task_data,
                "priority": priority
            },
            priority
        )

        logger.info(f"Task {task_id} assigned to {member.name}")

    async def task_completed(self, agent_id: str, task_id: str, result: Any):
        """任务完成"""
        if task_id in self.task_registry:
            self.task_registry[task_id]["status"] = "completed"
            self.task_registry[task_id]["result"] = result

        member = self.members.get(agent_id)
        if member:
            member.state = AgentState.IDLE
            member.current_task = None
            member.completed_tasks.append(task_id)

        # 通知 team lead
        if self.team_lead:
            await self.team_lead.mailbox.send_to(
                self.team_lead.agent_id,
                MessageType.RESULT,
                {
                    "task_id": task_id,
                    "result": result,
                    "from_agent": agent_id
                }
            )

        logger.info(f"Task {task_id} completed by {agent_id}")

    async def start_teamwork(self, task: Dict) -> Dict:
        """开始团队协作"""
        self._running = True

        if not self.team_lead:
            raise ValueError("Team needs a coordinator (Team Lead)")

        # Team Lead 分析任务并分解
        await self.team_lead.mailbox.send_to(
            self.team_lead.agent_id,
            MessageType.TASK,
            {"task": task, "type": "teamwork"}
        )

        # 启动所有成员的消息处理循环
        tasks = []
        for member in self.members.values():
            if member.mailbox:
                task = asyncio.create_task(
                    self._member_message_loop(member)
                )
                tasks.append(task)

        # 等待团队工作完成
        await asyncio.gather(*tasks, return_exceptions=True)

        self._running = False

        # 汇总结果
        return self._aggregate_results()

    async def _member_message_loop(self, member: TeamMember):
        """成员消息处理循环"""
        logger.info(f"{member.name} started message loop")

        while self._running:
            try:
                message = await member.mailbox.receive(timeout=1.0)

                if message:
                    await self._handle_message(member, message)

            except Exception as e:
                logger.error(f"Error in {member.name}'s message loop: {e}")

        logger.info(f"{member.name} stopped message loop")

    async def _handle_message(self, member: TeamMember, message: AgentMessage):
        """处理消息"""
        if message.type == MessageType.TASK:
            await self._handle_task(member, message)
        elif message.type == MessageType.QUERY:
            await self._handle_query(member, message)
        elif message.type == MessageType.APPROVAL:
            await self._handle_approval(member, message)

    async def _handle_task(self, member: TeamMember, message: AgentMessage):
        """处理任务消息"""
        task_id = message.content["task_id"]
        task_data = message.content["task_data"]

        try:
            # 执行任务（这里需要调用实际的 Agent）
            result = await self._execute_task(member, task_data)

            # 任务完成
            await self.task_completed(member.agent_id, task_id, result)

        except Exception as e:
            logger.error(f"Task {task_id} failed: {e}")
            await self._handle_task_failure(member, task_id, str(e))

    async def _execute_task(self, member: TeamMember, task_data: Dict) -> Any:
        """执行任务"""
        # 模拟任务执行
        await asyncio.sleep(0.1)
        return {"status": "success", "member": member.name, "data": task_data}

    async def _handle_task_failure(self, member: TeamMember, task_id: str, error: str):
        """处理任务失败"""
        self.task_registry[task_id]["status"] = "failed"
        self.task_registry[task_id]["error"] = error
        member.state = AgentState.IDLE
        member.current_task = None

    def _aggregate_results(self) -> Dict:
        """汇总结果"""
        completed = []
        failed = []

        for task_id, task_info in self.task_registry.items():
            if task_info["status"] == "completed":
                completed.append(task_info)
            elif task_info["status"] == "failed":
                failed.append(task_info)

        return {
            "team": self.team_name,
            "total_tasks": len(self.task_registry),
            "completed": len(completed),
            "failed": len(failed),
            "results": [t["result"] for t in completed],
            "errors": [t["error"] for t in failed]
        }
```

### 8.5 协调器实现

```python
class TeamCoordinator:
    """团队协调器"""

    def __init__(self, team_manager: TeamManager):
        self.team_manager = team_manager
        self.task_queue: List[Dict] = []

    async def coordinate(self, task: Dict) -> Dict:
        """协调团队完成任务"""
        # 1. 分析任务
        subtasks = self._decompose_task(task)

        # 2. 按依赖关系分配任务
        await self._allocate_tasks(subtasks)

        # 3. 监控进度
        await self._monitor_progress()

        # 4. 汇总结果
        return self._aggregate_team_results()

    def _decompose_task(self, task: Dict) -> List[Dict]:
        """分解任务"""
        task_type = task.get("type", "unknown")

        if task_type == "code_review":
            return [
                {
                    "subtask_id": "security_review",
                    "type": "security",
                    "assignee_type": RoleType.SPECIALIST,
                    "data": {"focus": "security"}
                },
                {
                    "subtask_id": "performance_review",
                    "type": "performance",
                    "assignee_type": RoleType.SPECIALIST,
                    "data": {"focus": "performance"}
                },
                {
                    "subtask_id": "style_review",
                    "type": "style",
                    "assignee_type": RoleType.REVIEWER,
                    "data": {"focus": "style"}
                }
            ]

        return []

    async def _allocate_tasks(self, subtasks: List[Dict]):
        """分配任务"""
        for subtask in subtasks:
            assignee_type = subtask["assignee_type"]
            members = self.team_manager.get_members_by_role(assignee_type)

            if members:
                # 选择空闲的成员
                assignee = self._select_idle_member(members)

                if assignee:
                    await self.team_manager.assign_task(
                        task_id=subtask["subtask_id"],
                        assignee=assignee.agent_id,
                        task_data=subtask["data"]
                    )

    def _select_idle_member(self, members: List[TeamMember]) -> Optional[TeamMember]:
        """选择空闲成员"""
        for member in members:
            if member.state == AgentState.IDLE:
                return member
        return members[0] if members else None

    async def _monitor_progress(self):
        """监控进度"""
        while True:
            all_completed = True

            for task_id, task_info in self.team_manager.task_registry.items():
                if task_info["status"] not in ["completed", "failed"]:
                    all_completed = False
                    await asyncio.sleep(0.5)
                    break

            if all_completed:
                break

    def _aggregate_team_results(self) -> Dict:
        """汇总团队结果"""
        return self.team_manager._aggregate_results()
```

### 8.6 冲突解决机制

#### 冲突类型

```python
class ConflictType(Enum):
    """冲突类型"""
    RESOURCE_CONTENTION = "resource_contention"  # 资源竞争
    TASK_OVERLAP = "task_overlap"               # 任务重叠
    RESULT_CONFLICT = "result_conflict"          # 结果冲突
    PRIORITY_CONFLICT = "priority_conflict"      # 优先级冲突
    DEPENDENCY_CYCLE = "dependency_cycle"        # 依赖循环
```

#### 冲突解决策略

```python
from typing import List, Tuple


class ConflictResolver:
    """冲突解决器"""

    def __init__(self):
        self.strategies = {
            ConflictType.RESOURCE_CONTENTION: self._resolve_resource,
            ConflictType.TASK_OVERLAP: self._resolve_task_overlap,
            ConflictType.RESULT_CONFLICT: self._resolve_result_conflict,
            ConflictType.PRIORITY_CONFLICT: self._resolve_priority,
        }

    async def resolve(self, conflict: Dict) -> Dict:
        """解决冲突"""
        conflict_type = ConflictType(conflict["type"])

        resolver = self.strategies.get(conflict_type)
        if resolver:
            return await resolver(conflict)

        return {"resolution": "manual_review_required"}

    async def _resolve_resource(self, conflict: Dict) -> Dict:
        """解决资源竞争"""
        # 策略1：优先级决定
        # 策略2：先到先得
        # 策略3：时间片轮转

        agents = conflict["agents"]
        resource = conflict["resource"]
        priority_a = conflict.get("priority_a", 1)
        priority_b = conflict.get("priority_b", 1)

        if priority_a > priority_b:
            winner = agents[0]
            loser = agents[1]
        elif priority_b > priority_a:
            winner = agents[1]
            loser = agents[0]
        else:
            # 优先级相同，使用 ID 排序
            winner, loser = sorted(agents)

        return {
            "resolution": "priority_based",
            "winner": winner,
            "loser": loser,
            "resource": resource,
            "action": "delayed"
        }

    async def _resolve_task_overlap(self, conflict: Dict) -> Dict:
        """解决任务重叠"""
        task1 = conflict["task1"]
        task2 = conflict["task2"]

        # 策略：合并任务，避免重复劳动
        return {
            "resolution": "task_merge",
            "merged_task": {
                "id": f"{task1['id']}+{task2['id']}",
                "agents": [task1["agent"], task2["agent"]],
                "scope": "combined"
            },
            "action": "collaborate"
        }

    async def _resolve_result_conflict(self, conflict: Dict) -> Dict:
        """解决结果冲突"""
        results = conflict["results"]

        # 策略1：投票决定
        # 策略2：信任度加权
        # 策略3：重新验证

        # 使用信任度加权
        weighted_votes = {}
        for result in results:
            agent = result["agent"]
            trust = result.get("trust", 1.0)
            vote = result["result"]

            if vote not in weighted_votes:
                weighted_votes[vote] = 0
            weighted_votes[vote] += trust

        winner = max(weighted_votes, key=weighted_votes.get)

        return {
            "resolution": "trust_weighted_vote",
            "winning_result": winner,
            "confidence": weighted_votes[winner] / sum(weighted_votes.values())
        }

    async def _resolve_priority(self, conflict: Dict) -> Dict:
        """解决优先级冲突"""
        # 高优先级任务获得资源
        tasks = sorted(
            [conflict["task1"], conflict["task2"]],
            key=lambda t: t.get("priority", 0),
            reverse=True
        )

        return {
            "resolution": "priority_order",
            "order": [t["id"] for t in tasks]
        }
```

### 8.7 团队绩效评估

```python
from dataclasses import dataclass
from typing import Dict, List
import time


@dataclass
class PerformanceMetrics:
    """性能指标"""
    agent_id: str
    tasks_completed: int = 0
    tasks_failed: int = 0
    total_time: float = 0
    quality_score: float = 0.0
    collaboration_score: float = 0.0
    efficiency: float = 0.0


class TeamPerformanceEvaluator:
    """团队绩效评估器"""

    def __init__(self, team_manager: TeamManager):
        self.team_manager = team_manager
        self.metrics: Dict[str, PerformanceMetrics] = {}
        self.task_durations: Dict[str, float] = {}
        self.task_start_times: Dict[str, float] = {}

    def start_tracking(self, task_id: str, agent_id: str):
        """开始追踪任务"""
        self.task_start_times[task_id] = time.time()

    def end_tracking(self, task_id: str, success: bool):
        """结束追踪任务"""
        if task_id in self.task_start_times:
            duration = time.time() - self.task_start_times[task_id]
            self.task_durations[task_id] = duration
            del self.task_start_times[task_id]

    def evaluate_member(self, agent_id: str) -> PerformanceMetrics:
        """评估成员绩效"""
        member = self.team_manager.get_member(agent_id)

        if not member:
            return PerformanceMetrics(agent_id=agent_id)

        # 计算指标
        tasks_completed = len(member.completed_tasks)
        tasks_failed = sum(
            1 for t in self.team_manager.task_registry.values()
            if t.get("assignee") == agent_id and t["status"] == "failed"
        )

        total_time = sum(
            self.task_durations.get(t, 0)
            for t in member.completed_tasks
        )

        # 效率 = 完成任务数 / 总时间
        efficiency = tasks_completed / total_time if total_time > 0 else 0

        metrics = PerformanceMetrics(
            agent_id=agent_id,
            tasks_completed=tasks_completed,
            tasks_failed=tasks_failed,
            total_time=total_time,
            efficiency=efficiency
        )

        self.metrics[agent_id] = metrics
        return metrics

    def evaluate_team(self) -> Dict:
        """评估团队整体绩效"""
        if not self.team_manager.team_lead:
            return {}

        # 评估所有成员
        for agent_id in self.team_manager.members:
            self.evaluate_member(agent_id)

        # 计算团队指标
        total_tasks = len(self.team_manager.task_registry)
        completed_tasks = sum(
            1 for t in self.team_manager.task_registry.values()
            if t["status"] == "completed"
        )

        return {
            "team_name": self.team_manager.team_name,
            "total_members": len(self.team_manager.members),
            "total_tasks": total_tasks,
            "completed_tasks": completed_tasks,
            "completion_rate": completed_tasks / total_tasks if total_tasks > 0 else 0,
            "member_metrics": {
                agent_id: {
                    "tasks_completed": m.tasks_completed,
                    "efficiency": m.efficiency
                }
                for agent_id, m in self.metrics.items()
            }
        }

    def generate_report(self) -> str:
        """生成绩效报告"""
        team_eval = self.evaluate_team()

        report = f"""# 团队绩效报告

## 团队概况

- 团队名称：{team_eval.get('team_name')}
- 成员数量：{team_eval.get('total_members')}
- 总任务数：{team_eval.get('total_tasks')}
- 完成任务：{team_eval.get('completed_tasks')}
- 完成率：{team_eval.get('completion_rate', 0) * 100:.1f}%

## 成员绩效

| 成员 | 完成任务 | 失败任务 | 效率 |
|------|----------|----------|------|
"""

        for agent_id, metrics in self.metrics.items():
            member = self.team_manager.get_member(agent_id)
            name = member.name if member else agent_id
            report += f"| {name} | {metrics.tasks_completed} | {metrics.tasks_failed} | {metrics.efficiency:.2f} |\n"

        return report
```
