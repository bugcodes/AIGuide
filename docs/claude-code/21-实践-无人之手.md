### 19.1 CI/CD 基础

#### 什么是 CI/CD？

```
CI/CD = 持续集成 / 持续交付

CI（持续集成）：
- 开发者频繁提交代码
- 自动构建和测试
- 快速发现集成问题

CD（持续交付）：
- 自动部署到测试环境
- 自动部署到生产环境
- 减少人工干预
```

#### CI/CD 流水线

```
典型流水线：

1. Code Checkout
   └─> 拉取代码

2. Build
   └─> 编译/打包

3. Test
   ├─> 单元测试
   ├─> 集成测试
   └─> E2E 测试

4. Security Scan
   ├─> 依赖检查
   └─> 代码扫描

5. Deploy
   ├─> Staging 环境
   └─> Production 环境

6. Monitor
   └─> 监控和告警
```

### 19.2 GitHub Actions 集成

#### 创建工作流文件

```yaml
# .github/workflows/ci.yml

name: CI Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  quality-checks:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          pip install -r requirements.txt
          pip install pytest pytest-cov ruff bandit

      - name: Run linting
        run: ruff check src/

      - name: Run tests
        run: pytest tests/ -v --cov=src --cov-report=xml

      - name: Security scan
        run: bandit -r src/

      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          file: ./coverage.xml

  build:
    needs: quality-checks
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Build Docker image
        run: docker build -t myapp:${{ github.sha }} .

      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_TOKEN }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker push myapp:${{ github.sha }}
```

### 19.3 Claude Code 作为 CI Step

```yaml
# .github/workflows/claude-review.yml

name: Claude Code Review

on:
  pull_request:
    types: [opened, synchronize]

jobs:
  claude-review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Run Claude Code Review
        uses: claude-code/github-action@v1
        env:
          CLAUDE_API_KEY: ${{ secrets.CLAUDE_API_KEY }}
        with:
          commands: |
            /review --pr=${{ github.event.pull_request.number }}
            /security-audit --focus=critical,high
            /generate-report --format=github-annotation
```

### 19.4 智能发布管理

```python
class IntelligentReleaseManager:
    """智能发布管理器"""

    def __init__(self):
        self.release_branch = "main"
        self.hotfix_branch = "hotfix"
        self.changelog_generator = ChangelogGenerator()
        self.version_bumper = VersionBumper()

    async def create_release(
        self,
        current_version: str,
        release_type: str  # major, minor, patch
    ) -> Release:
        """创建新版本"""
        # 1. 计算新版本号
        new_version = self.version_bumper.bump(
            current_version,
            release_type
        )

        # 2. 生成变更日志
        changelog = await self.changelog_generator.generate(
            since=current_version
        )

        # 3. 创建发布分支
        await self.git.create_branch(
            f"release/{new_version}",
            from_branch=self.release_branch
        )

        # 4. 更新版本文件
        await self._update_version_file(new_version)

        # 5. 提交并推送
        await self.git.commit(
            f"chore: release version {new_version}\n\n{Changelog}",
            branch=f"release/{new_version}"
        )

        # 6. 创建 PR
        pr = await self.git.create_pr(
            title=f"Release {new_version}",
            body=self._generate_pr_body(changelog),
            from_branch=f"release/{new_version}",
            to_branch=self.release_branch
        )

        return Release(
            version=new_version,
            changelog=changelog,
            pr_id=pr.id
        )

    async def deploy_to_production(self, release: Release):
        """部署到生产环境"""
        # 1. 等待 CI 通过
        await self.wait_for_ci(release.pr_id)

        # 2. 合并 PR
        await self.git.merge_pr(release.pr_id)

        # 3. 创建 Git Tag
        await self.git.create_tag(
            f"v{release.version}",
            message=f"Release {release.version}"
        )

        # 4. 部署
        await self.deployer.deploy(
            environment="production",
            version=release.version
        )

        # 5. 发送通知
        await self.notifier.send(
            f":rocket: Version {release.version} deployed!"
        )


class ChangelogGenerator:
    """变更日志生成器"""

    async def generate(self, since: str) -> str:
        """生成变更日志"""
        commits = await self.git.get_commits(since=since)

        changes = {
            "added": [],
            "changed": [],
            "fixed": [],
            "removed": [],
            "security": []
        }

        for commit in commits:
            message = commit.message.lower()

            if message.startswith("feat"):
                changes["added"].append(commit)
            elif message.startswith("fix"):
                changes["fixed"].append(commit)
            elif message.startswith("refactor"):
                changes["changed"].append(commit)
            elif message.startswith("perf"):
                changes["changed"].append(commit)
            elif "security" in message:
                changes["security"].append(commit)

        return self._format_changelog(changes)

    def _format_changelog(self, changes: dict) -> str:
        """格式化变更日志"""
        sections = []

        if changes["added"]:
            sections.append("### Added\n")
            for c in changes["added"]:
                sections.append(f"- {c.message}\n")

        if changes["fixed"]:
            sections.append("### Fixed\n")
            for c in changes["fixed"]:
                sections.append(f"- {c.message}\n")

        return "\n".join(sections)
```

### 19.5 部署策略

```python
class DeploymentStrategy:
    """部署策略"""

    @staticmethod
    async def blue_green(deployer: Deployer, version: str):
        """蓝绿部署"""
        # 1. 部署到绿色环境
        await deployer.deploy(
            environment="green",
            version=version
        )

        # 2. 运行烟雾测试
        if not await deployer.smoke_test("green"):
            raise DeploymentError("Smoke test failed")

        # 3. 切换流量
        await deployer.switch_traffic("green")

        # 4. 监控
        await deployer.monitor(environment="green", duration=300)

        # 5. 关闭蓝色环境
        if deployer.is_healthy("blue"):
            await deployer.shutdown("blue")

    @staticmethod
    async def canary(deployer: Deployer, version: str, traffic_percent: int = 10):
        """金丝雀部署"""
        # 1. 部署 10% 流量
        await deployer.deploy(
            environment="canary",
            version=version,
            replicas=1
        )

        await deployer.configure_traffic_split(
            original=100 - traffic_percent,
            canary=traffic_percent
        )

        # 2. 监控金丝雀指标
        metrics = await deployer.monitor(
            environment="canary",
            duration=600
        )

        # 3. 逐步增加流量
        for percent in [25, 50, 100]:
            if metrics.is_healthy():
                await deployer.configure_traffic_split(
                    original=100 - percent,
                    canary=percent
                )
            else:
                # 回滚
                await deployer.rollback()
                raise DeploymentError("Canary metrics degraded")

    @staticmethod
    async def rolling(deployer: Deployer, version: str):
        """滚动部署"""
        replicas = await deployer.get_replica_count()

        # 逐个替换
        for i in range(replicas):
            await deployer.deploy_single(version=version)

            if not await deployer.is_healthy():
                raise DeploymentError(f"Replica {i} failed")

            await asyncio.sleep(30)  # 等待稳定
```

### 19.6 回滚机制

```python
class RollbackManager:
    """回滚管理器"""

    def __init__(self):
        self.deployment_history = DeploymentHistory()

    async def rollback(
        self,
        environment: str,
        reason: str = None
    ) -> RollbackResult:
        """执行回滚"""
        # 1. 获取上一个稳定版本
        last_good = await self.deployment_history.get_last_good(
            environment
        )

        if not last_good:
            raise RollbackError("No good deployment found")

        # 2. 记录回滚原因
        await self._log_rollback_reason(reason)

        # 3. 执行回滚
        await self.deployer.deploy(
            environment=environment,
            version=last_good.version,
            force=True
        )

        # 4. 验证
        if await self.deployer.is_healthy(environment):
            return RollbackResult(
                success=True,
                from_version=last_good.version,
                to_version=last_good.version
            )

        raise RollbackError("Rollback verification failed")

    async def auto_rollback_on_failure(
        self,
        environment: str,
        check_interval: int = 60
    ):
        """自动回滚"""
        while True:
            metrics = await self.deployer.get_metrics(environment)

            if metrics.error_rate > 0.05:  # 5% 错误率阈值
                print(f"High error rate detected: {metrics.error_rate}")
                await self.rollback(
                    environment,
                    reason=f"Auto-rollback: error rate {metrics.error_rate}"
                )
                break

            if metrics.latency_p99 > 5000:  # 5s 延迟阈值
                print(f"High latency detected: {metrics.latency_p99}")
                await self.rollback(
                    environment,
                    reason=f"Auto-rollback: latency {metrics.latency_p99}ms"
                )
                break

            await asyncio.sleep(check_interval)
```

### 19.7 监控告警

```python
class DeploymentMonitor:
    """部署监控"""

    def __init__(self):
        self.metrics_collector = MetricsCollector()
        self.alert_manager = AlertManager()

    async def monitor_deployment(
        self,
        deployment_id: str,
        duration: int = 600
    ) -> MonitoringReport:
        """监控部署"""
        start_time = time.time()
        metrics_history = []

        while time.time() - start_time < duration:
            # 收集指标
            metrics = await self.metrics_collector.collect()

            # 检查告警条件
            alerts = self._check_alerts(metrics)

            if alerts:
                await self.alert_manager.send_alerts(alerts)

            metrics_history.append(metrics)

            await asyncio.sleep(30)

        return MonitoringReport(
            deployment_id=deployment_id,
            metrics=metrics_history,
            alerts=alerts,
            summary=self._generate_summary(metrics_history)
        )

    def _check_alerts(self, metrics: Metrics) -> List[Alert]:
        """检查告警"""
        alerts = []

        if metrics.error_rate > 0.01:
            alerts.append(Alert(
                level="warning",
                message=f"Error rate: {metrics.error_rate:.2%}"
            ))

        if metrics.error_rate > 0.05:
            alerts.append(Alert(
                level="critical",
                message=f"Critical error rate: {metrics.error_rate:.2%}"
            ))

        if metrics.latency_p99 > 2000:
            alerts.append(Alert(
                level="warning",
                message=f"High latency P99: {metrics.latency_p99}ms"
            ))

        return alerts
```
