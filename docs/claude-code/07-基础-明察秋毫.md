### 5.2 只读代理的设计原则

#### 为什么只读很重要？

```
安全审计的核心原则：不造成额外伤害

只读设计确保：
1. 审计过程不会引入新的漏洞
2. 审计过程不会破坏现有代码
3. 审计过程可以被审计（审计的审计）
4. 审计结果基于代码当前状态，不受审计过程影响
```

#### 只读代理的架构设计

```python
class ReadOnlySecurityAuditAgent:
    """
    只读型安全审计代理

    核心原则：
    - 只能读取文件和搜索代码
    - 禁止任何写入操作
    - 禁止执行任何可能修改系统的命令
    """

    def __init__(self):
        # 允许的工具：只读操作
        self.allowed_tools = {
            "read_file": self._safe_read_file,
            "glob": self._safe_glob_files,
            "grep": self._safe_search_in_files,
            "search_files": self._safe_content_search,
        }

        # 禁止的工具：可能修改系统的操作
        self.forbidden_tools = {
            "write_file",      # 禁止写入文件
            "edit",            # 禁止编辑文件
            "bash",           # 禁止执行命令
            "terminal",       # 禁止终端操作
            "delete",         # 禁止删除
        }

        # 审计约束
        self.constraints = [
            "审计过程不会修改任何文件",
            "审计过程不会执行任何命令",
            "审计结果仅基于静态代码分析",
            "审计报告不包含敏感信息",
        ]

    def _safe_read_file(self, path: str) -> str:
        """安全读取文件"""
        # 验证路径安全性
        resolved_path = self._validate_path(path)

        # 只允许读取代码文件
        allowed_extensions = {
            '.py', '.js', '.ts', '.java', '.go', '.rs',
            '.c', '.cpp', '.h', '.hpp', '.cs', '.rb',
            '.php', '.swift', '.kt', '.scala', '.sql'
        }

        if not any(resolved_path.endswith(ext) for ext in allowed_extensions):
            raise SecurityAuditError(f"不支持的文件类型: {resolved_path}")

        return self._read_file_content(resolved_path)
```

#### 漏洞 1：SQL 注入

**原理：**

```
正常查询：
SELECT * FROM users WHERE id = 1

注入攻击：
SELECT * FROM users WHERE id = 1 OR 1=1

后果：
- 数据泄露：攻击者获取整个数据库内容
- 数据篡改：修改或删除数据
- 身份绕过：绕过登录验证
- 远程执行：在某些条件下执行系统命令
```

**代码示例：**

```python
# 脆弱的代码（Python + SQLAlchemy）
def get_user_vulnerable(user_id):
    # 危险：直接拼接 SQL
    query = f"SELECT * FROM users WHERE id = {user_id}"
    result = db.execute(query)
    return result

# 攻击示例：
# user_id = "1 OR 1=1"  -> 获取所有用户
# user_id = "1; DROP TABLE users; --"  -> 删除表
```

```python
# 安全的代码（使用参数化查询）
def get_user_safe(user_id):
    # 安全：参数化查询
    query = "SELECT * FROM users WHERE id = :user_id"
    result = db.execute(query, {"user_id": user_id})
    return result
```

**SQL 注入检测器实现：**

```python
class SQLInjectionDetector:
    """SQL 注入漏洞检测器"""

    def __init__(self):
        self.severity = "critical"
        self.cwe_id = "CWE-89"

        # 危险模式
        self.dangerous_patterns = [
            {
                "name": "String concatenation in SQL",
                "pattern": r'["\']\s*(SELECT|INSERT|UPDATE|DELETE|DROP|UNION).*\{.*\}',
                "severity": "critical"
            },
            {
                "name": "f-string SQL query",
                "pattern": r'f["\'].*\%s.*["\']|f["\'].*\{.*\}.*["\']',
                "severity": "critical"
            },
            {
                "name": "format() in SQL",
                "pattern": r'\.format\(.*\).*(SELECT|INSERT|UPDATE|DELETE)',
                "severity": "critical"
            },
            {
                "name": "execute with string formatting",
                "pattern": r'\.execute\([^)]*%[^)]*\)',
                "severity": "critical"
            },
            {
                "name": "raw query execution",
                "pattern": r'\.raw\([^)]*\)',
                "severity": "high"
            }
        ]

        # 安全模式（用于排除误报）
        self.safe_patterns = [
            r'cursor\.execute\([^)]*,\s*\[',  # 参数化查询
            r'cursor\.execute\([^)]*,\s*\{',  # 命名参数
            r'\$\d+',                          # 位置参数
            r'SELECT \* FROM \w+ WHERE \w+ = \?',
        ]

    def detect(self, code: str, file_path: str) -> List[Vulnerability]:
        """检测代码中的 SQL 注入漏洞"""
        vulnerabilities = []

        for pattern_info in self.dangerous_patterns:
            matches = self._find_pattern(code, pattern_info["pattern"])

            for match in matches:
                # 检查是否有对应的安全模式（排除误报）
                if not self._is_false_positive(match, code):
                    vulnerability = Vulnerability(
                        type="sql_injection",
                        severity=pattern_info["severity"],
                        file_path=file_path,
                        line_number=match["line"],
                        code_snippet=match["code"],
                        description=pattern_info["name"],
                        cwe_id=self.cwe_id,
                        remediation=self._get_remediation("sql_injection")
                    )
                    vulnerabilities.append(vulnerability)

        return vulnerabilities

    def _is_false_positive(self, match: Dict, code: str) -> bool:
        """检查是否为误报"""
        context = self._get_context(code, match["line"], lines=3)

        for safe_pattern in self.safe_patterns:
            if re.search(safe_pattern, context):
                return True

        return False

    def _get_remediation(self, vuln_type: str) -> str:
        """获取修复建议"""
        remediations = {
            "sql_injection": """
                修复方法：

                1. 使用参数化查询（推荐）

                # Python (SQLAlchemy)
                cursor.execute(
                    "SELECT * FROM users WHERE id = :user_id",
                    {"user_id": user_id}
                )

                # Python (psycopg2)
                cursor.execute(
                    "SELECT * FROM users WHERE id = %s",
                    (user_id,)
                )

                2. 使用 ORM（最安全）

                # SQLAlchemy ORM
                result = session.query(User).filter_by(id=user_id).first()

                3. 输入验证（辅助措施）

                - 验证输入类型（应为整数）
                - 使用白名单验证
            """
        }
        return remediations.get(vuln_type, "")
```

#### 漏洞 2：认证失效

**原理：**

```
认证失效指：
1. 会话 ID 暴露在 URL 中
2. 会话 ID 未正确失效
3. 密码策略太弱
4. 多因素认证未启用
5. 凭据存储不安全

后果：
- 账户被盗
- 未授权访问
- 身份冒充
```

**代码示例：**

```python
# 脆弱的代码：会话管理不当
class VulnerableAuth:
    def login(self, username, password):
        user = db.query(User).filter_by(username=username).first()

        if user and user.password == password:  # 明文密码比较
            # 危险：会话 ID 暴露在 URL
            session_id = generate_session_id()
            self.sessions[session_id] = user.id
            return f"/dashboard?session={session_id}"  # 不安全

        return "/login?error=invalid"

# 安全的代码
class SecureAuth:
    def login(self, username, password):
        user = db.query(User).filter_by(username=username).first()

        if user and bcrypt.checkpw(password.encode(), user.password_hash):
            # 安全：使用安全的会话管理
            session = Session.create(user_id=user.id)
            response = redirect("/dashboard")
            response.set_cookie(
                "session_id",
                session.id,
                httponly=True,  # 防止 XSS 访问
                secure=True,     # 仅 HTTPS
                samesite="Lax"  # CSRF 保护
            )
            return response

        return "/login?error=invalid"
```

#### 漏洞 3：敏感数据泄露

**原理：**

```
敏感数据包括：
- 个人身份信息（PII）：姓名、地址、电话、身份证
- 财务信息：信用卡号、银行账户
- 健康信息（PHI）：医疗记录
- 登录凭据：用户名、密码、API 密钥

泄露途径：
1. 日志中打印敏感数据
2. 错误消息中包含敏感信息
3. 未加密的数据传输
4. 硬编码的密钥
5. 错误的存储权限
```

**代码示例：**

```python
# 脆弱的代码：敏感数据泄露
class DataHandlingVulnerable:
    def process_payment(self, card_number, cvv):
        # 危险：日志中打印敏感信息
        logger.info(f"Processing payment: card={card_number}, cvv={cvv}")

        # 危险：返回敏感数据
        return {
            "status": "success",
            "card_last_4": card_number[-4:],
            "cvv": cvv  # 不应该返回 CVV
        }

# 安全的代码
class DataHandlingSecure:
    def process_payment(self, card_number, cvv):
        # 安全：使用掩码日志
        masked_card = f"****-****-****-{card_number[-4:]}"
        logger.info(f"Processing payment: card={masked_card}")

        # 安全：只返回必要信息
        return {
            "status": "success",
            "card_last_4": card_number[-4:]
            # 不返回 CVV
        }
```

**敏感数据检测器实现：**

```python
class SensitiveDataExposureDetector:
    """敏感数据泄露检测器"""

    def __init__(self):
        self.severity = "critical"
        self.cwe_id = "CWE-200"

        # 敏感数据模式
        self.sensitive_patterns = [
            {
                "name": "Hardcoded API key",
                "pattern": r'api[_\w]*[_\s]*key\s*=\s*["\'][a-zA-Z0-9_-]{20,}["\']',
                "severity": "critical"
            },
            {
                "name": "Hardcoded password",
                "pattern": r'(password|passwd|pwd|secret)\s*=\s*["\'][^"\']{4,}["\']',
                "severity": "critical"
            },
            {
                "name": "Credit card number",
                "pattern": r'\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b',
                "severity": "critical"
            },
            {
                "name": "Logging sensitive data",
                "pattern": r'log(ger)?\.(info|debug|warn)\([^)]*(password|secret|key|token).*\)',
                "severity": "high"
            },
            {
                "name": "Private key in code",
                "pattern": r'-----BEGIN\s+(RSA|DSA|EC|OPENSSH)\s+PRIVATE KEY-----',
                "severity": "critical"
            }
        ]

        # 排除模式（误报）
        self.exclusion_patterns = [
            r'example\.com',      # 示例域名
            r'your_api_key',      # 占位符
            r'<placeholders>',     # 占位符
            r'os\.environ\.get',  # 环境变量读取
        ]

    def detect(self, code: str, file_path: str) -> List[Vulnerability]:
        """检测敏感数据泄露"""
        vulnerabilities = []

        for pattern_info in self.sensitive_patterns:
            matches = self._find_pattern(code, pattern_info["pattern"])

            for match in matches:
                # 排除误报
                if self._is_excluded(match["code"]):
                    continue

                vulnerability = Vulnerability(
                    type="sensitive_data_exposure",
                    severity=pattern_info["severity"],
                    file_path=file_path,
                    line_number=match["line"],
                    code_snippet=self._mask_sensitive(match["code"]),
                    description=pattern_info["name"],
                    cwe_id=self.cwe_id,
                    remediation=self._get_remediation(pattern_info["name"])
                )
                vulnerabilities.append(vulnerability)

        return vulnerabilities

    def _mask_sensitive(self, code: str) -> str:
        """掩码敏感信息"""
        # 掩码 API 密钥
        code = re.sub(
            r'([a-zA-Z0-9_-]{20,})',
            r'\1 -> [MASKED]',
            code
        )
        return code
```

#### 漏洞 4：跨站脚本（XSS）

**原理：**

```
XSS 攻击类型：
1. 反射型 XSS：恶意脚本来自 URL 参数
2. 存储型 XSS：恶意脚本存储在数据库中
3. DOM 型 XSS：恶意脚本通过 DOM 操作注入

后果：
- 窃取 Cookie/Session
- 篡改页面内容
- 键盘记录
- 重定向到钓鱼网站
```

**代码示例：**

```python

def search():

    # 危险：直接渲染用户输入

# /search?q=<script>alert(document.cookie)</script>

# 安全的代码

    query = request.args.get('q', '')

    # 或手动转义

    safe_query = escape(query)

#### 漏洞 5：访问控制失效

**原理：**

```
访问控制失效指：
1. 垂直提权：普通用户获得管理员权限
2. 水平提权：用户访问其他用户的数据
3. 路径遍历：访问目录外的文件
4. IDOR：直接对象引用

后果：
- 未授权访问
- 数据泄露
- 权限绕过
```

**代码示例：**

```python
# 脆弱的代码：访问控制失效
@app.route('/user/<int:user_id>/delete', methods=['POST'])
def delete_user(user_id):
    # 危险：没有检查当前用户权限
    User.query.filter_by(id=user_id).delete()
    return "User deleted"

# 安全的代码
@app.route('/user/<int:user_id>/delete', methods=['POST'])
@login_required
@admin_required  # 需要管理员权限
def delete_user(user_id):
    # 安全：权限检查
    current_user = get_current_user()
    if not current_user.is_admin:
        abort(403)

    if current_user.id != user_id and not current_user.is_super_admin:
        abort(403)  # 不能删除其他管理员

    User.query.filter_by(id=user_id).delete()
    return "User deleted"
```

**访问控制检测器实现：**

```python
class AccessControlDetector:
    """访问控制漏洞检测器"""

    def __init__(self):
        self.severity = "high"
        self.cwe_id = "CWE-639"

        self.rules = [
            {
                "name": "Missing authorization decorator",
                "pattern": r'@app\.route\([^)]*\)\s*\n\s*def\s+\w+\([^)]*\):\s*\n\s*(?!.*@(login_required|admin_required|permission_required))',
                "severity": "high",
                "multiline": True
            },
            {
                "name": "Direct object reference",
                "pattern": r'\.query\.filter_by\(id\s*=\s*\w+\)\s*\n\s*return',
                "severity": "medium",
                "description": "可能存在直接对象引用问题"
            }
        ]

    def detect(self, code: str, file_path: str) -> List[Vulnerability]:
        """检测访问控制漏洞"""
        vulnerabilities = []

        for rule in self.rules:
            flags = re.MULTILINE
            if rule.get("multiline"):
                flags |= re.DOTALL

            matches = self._find_pattern(code, rule["pattern"], flags=flags)

            for match in matches:
                vulnerability = Vulnerability(
                    type="broken_access_control",
                    severity=rule["severity"],
                    file_path=file_path,
                    line_number=self._get_line_number(code, match["start"]),
                    code_snippet=match["code"],
                    description=rule["description"],
                    cwe_id=self.cwe_id,
                    remediation=self._get_remediation(rule["name"])
                )
                vulnerabilities.append(vulnerability)

        return vulnerabilities
```

class SecurityAuditReport:
    """安全审计报告"""

    def __init__(self):
        self.vulnerabilities: List[Vulnerability] = []
        self.scan_date = datetime.now()
        self.total_files_scanned = 0

    def add_vulnerability(self, vuln: Vulnerability):
        self.vulnerabilities.append(vuln)

    def get_summary(self) -> Dict:
        """获取漏洞统计摘要"""
        severity_counts = {}
        type_counts = {}

        for vuln in self.vulnerabilities:
            severity_counts[vuln.severity] = severity_counts.get(vuln.severity, 0) + 1
            type_counts[vuln.type] = type_counts.get(vuln.type, 0) + 1

        return {
            "total_vulnerabilities": len(self.vulnerabilities),
            "by_severity": severity_counts,
            "by_type": type_counts,
            "files_scanned": self.total_files_scanned,
            "scan_date": self.scan_date.isoformat()
        }

    def get_report_md(self) -> str:
        """生成 Markdown 格式报告"""
        summary = self.get_summary()

        report = f"""# 安全审计报告

## 概要

- 扫描时间：{self.scan_date.strftime('%Y-%m-%d %H:%M:%S')}
- 扫描文件数：{self.total_files_scanned}
- 发现漏洞总数：{summary['total_vulnerabilities']}

## 漏洞统计

### 按严重程度

| 严重程度 | 数量 |
|----------|------|
| Critical | {summary['by_severity'].get('critical', 0)} |
| High | {summary['by_severity'].get('high', 0)} |
| Medium | {summary['by_severity'].get('medium', 0)} |
| Low | {summary['by_severity'].get('low', 0)} |

### 按类型

| 类型 | 数量 |
|------|------|
"""

        for vuln_type, count in summary['by_type'].items():
            report += f"| {vuln_type} | {count} |\n"

        report += "\n## 漏洞详情\n\n"

        # 按严重程度排序
        severity_order = {"critical": 0, "high": 1, "medium": 2, "low": 3}
        sorted_vulns = sorted(
            self.vulnerabilities,
            key=lambda v: severity_order.get(v.severity, 4)
        )

        for vuln in sorted_vulns:
            report += f"""### {vuln.severity.upper()}: {vuln.type}

- **文件**: {vuln.file_path}:{vuln.line_number}
- **CWE**: {vuln.cwe_id}
- **描述**: {vuln.description}

**问题代码:**
```code
{vuln.code_snippet}
```

**修复建议:**
{vuln.remediation}

---

"""

        return report


class SecurityAuditAgent:
    """
    只读型安全审计子代理

    设计原则：
    1. 只使用只读工具（read_file, glob, grep）
    2. 不修改任何文件
    3. 不执行任何命令
    4. 只生成报告，不生成修复代码
    """

    def __init__(self, config: AuditConfig = None):
        self.config = config or AuditConfig()

        # 只使用只读工具
        self.tools = ["read_file", "glob", "grep"]

        # 约束
        self.constraints = [
            "禁止修改任何文件",
            "禁止执行任何命令",
            "禁止访问网络",
            "审计结果仅基于静态代码分析"
        ]

        # 漏洞检测器
        self.detectors = [
            SQLInjectionDetector(),
            XSSDetector(),
            AuthenticationVulnerabilityDetector(),
            SensitiveDataExposureDetector(),
            AccessControlDetector()
        ]

    async def audit(self, target_path: str) -> SecurityAuditReport:
        """执行安全审计"""
        report = SecurityAuditReport()

        # 1. 收集代码文件
        code_files = await self._collect_code_files(target_path)
        report.total_files_scanned = len(code_files)

        # 2. 并行运行所有检测器
        all_vulnerabilities = []

        for file_path in code_files:
            code = await self._read_file(file_path)

            for detector in self.detectors:
                try:
                    vulnerabilities = detector.detect(code, file_path)
                    all_vulnerabilities.extend(vulnerabilities)
                except Exception as e:
                    logger.warning(f"Detector {detector.__class__.__name__} failed on {file_path}: {e}")

        # 3. 添加到报告
        for vuln in all_vulnerabilities:
            report.add_vulnerability(vuln)

        return report

    async def _collect_code_files(self, target_path: str) -> List[str]:
        """收集需要审计的代码文件"""
        patterns = [
            "**/*.py",
            "**/*.js",
            "**/*.ts",
            "**/*.java",
            "**/*.go",
            "**/*.rb",
            "**/*.php"
        ]

        files = []
        for pattern in patterns:
            matched = await self._glob(pattern, target_path)
            files.extend(matched)

        return files
```

  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

  security-audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Run Security Audit
        run: |
          python -m pip install security-audit-tool
          security-audit scan ./src --format json --output report.json

      - name: Check for Critical Vulnerabilities
        run: |
          CRITICAL=$(jq '[.vulnerabilities[] | select(.severity == "critical")] | length' report.json)
          if [ "$CRITICAL" -gt 0 ]; then
            echo "Found $CRITICAL critical vulnerabilities"
            exit 1
          fi

      - name: Upload Report
        uses: actions/upload-artifact@v3
        with:
          name: security-audit-report
          path: report.json
