### 13.1 代码库探索方法论

#### 为什么需要系统化探索？

```
随机探索的问题：
- 效率低下
- 容易遗漏关键信息
- 无法形成整体认知

系统化探索的优势：
- 全面覆盖
- 效率更高
- 能发现隐藏关系
```

#### 探索框架

```python
class CodebaseExplorer:
    """代码库探索器"""

    def __init__(self, root_path: str):
        self.root_path = Path(root_path)
        self.cache = {}
        self.dependency_graph = {}

    async def explore(self) -> CodebaseMap:
        """全面探索代码库"""
        explorer = CodebaseExplorer(self.root_path)

        # Phase 1: 收集文件结构
        structure = await explorer._explore_structure()

        # Phase 2: 分析依赖关系
        dependencies = await explorer._analyze_dependencies(structure)

        # Phase 3: 识别代码模式
        patterns = await explorer._identify_patterns(structure)

        # Phase 4: 生成语义地图
        semantic_map = await explorer._generate_semantic_map(
            structure,
            dependencies,
            patterns
        )

        return semantic_map

    async def _explore_structure(self) -> StructureMap:
        """探索文件结构"""
        structure = StructureMap()

        # 收集所有代码文件
        code_files = list(self.root_path.rglob("*.py"))  # Python
        code_files.extend(self.root_path.rglob("*.js"))  # JavaScript
        code_files.extend(self.root_path.rglob("*.ts"))  # TypeScript
        code_files.extend(self.root_path.rglob("*.go"))  # Go

        # 按目录分组
        for file in code_files:
            relative_path = file.relative_to(self.root_path)
            structure.add_file(relative_path, file)

        return structure
```

### 13.2 代码库结构分析

```python
@dataclass
class StructureAnalysis:
    """结构分析结果"""
    total_files: int
    total_lines: int
    languages: Dict[str, int]  # 语言 -> 文件数
    directories: Dict[str, FileStats]  # 目录 -> 统计
    largest_files: List[FileInfo]  # 最大文件
    deepest_paths: List[Path]  # 最深路径


class StructureAnalyzer:
    """结构分析器"""

    def analyze(self, root_path: Path) -> StructureAnalysis:
        """分析代码库结构"""
        stats = {
            "total_files": 0,
            "total_lines": 0,
            "languages": defaultdict(int),
            "directories": defaultdict(FileStats),
            "file_sizes": []
        }

        for file in root_path.rglob("*"):
            if file.is_file() and self._is_code_file(file):
                stats["total_files"] += 1

                # 统计行数
                try:
                    lines = len(file.read_text().splitlines())
                    stats["total_lines"] += lines
                    stats["file_sizes"].append((file, lines))
                except:
                    pass

                # 统计语言
                ext = file.suffix
                stats["languages"][ext] += 1

                # 统计目录
                dir_path = file.parent
                stats["directories"][dir_path].file_count += 1
                stats["directories"][dir_path].total_lines += lines

        # 找出最大文件
        largest_files = sorted(
            stats["file_sizes"],
            key=lambda x: x[1],
            reverse=True
        )[:10]

        return StructureAnalysis(
            total_files=stats["total_files"],
            total_lines=stats["total_lines"],
            languages=dict(stats["languages"]),
            directories=dict(stats["directories"]),
            largest_files=[FileInfo(f, lines) for f, lines in largest_files],
            deepest_paths=self._find_deepest_paths(root_path)
        )

    def _find_deepest_paths(self, root: Path, max_depth: int = 10) -> List[Path]:
        """找出最深的路径"""
        deepest = []

        for file in root.rglob("*"):
            if file.is_file():
                depth = len(file.parts) - len(root.parts)
                if depth >= max_depth - 2:
                    deepest.append(file)

        return sorted(deepest, key=lambda p: len(p.parts), reverse=True)[:5]
```

### 13.3 依赖关系分析

```python
class DependencyAnalyzer:
    """依赖关系分析器"""

    def __init__(self):
        self.import_patterns = {
            "python": self._python_imports,
            "javascript": self._js_imports,
            "typescript": self._ts_imports,
        }

    def analyze(self, files: List[Path]) -> DependencyGraph:
        """分析文件间的依赖关系"""
        graph = DependencyGraph()

        for file in files:
            lang = self._detect_language(file)
            if lang in self.import_patterns:
                imports = self.import_patterns[lang](file)
                graph.add_node(str(file))

                for imported in imports:
                    graph.add_edge(str(file), imported)

        return graph

    def _python_imports(self, file: Path) -> List[str]:
        """提取 Python 导入"""
        content = file.read_text()
        imports = []

        # import xxx
        for match in re.finditer(r'^import\s+([\w.]+)', content, re.MULTILINE):
            imports.append(match.group(1))

        # from xxx import yyy
        for match in re.finditer(r'^from\s+([\w.]+)\s+import', content, re.MULTILINE):
            imports.append(match.group(1))

        return imports

    def _js_imports(self, file: Path) -> List[str]:
        """提取 JavaScript 导入"""
        content = file.read_text()
        imports = []

        # import xxx from 'yyy'
        for match in re.finditer(r"import\s+.*?\s+from\s+['\"]([^'\"]+)['\"]", content):
            imports.append(match.group(1))

        # require('xxx')
        for match in re.finditer(r"require\s*\(\s*['\"]([^'\"]+)['\"]", content):
            imports.append(match.group(1))

        return imports


class DependencyGraph:
    """依赖图"""

    def __init__(self):
        self.nodes: Set[str] = set()
        self.edges: Dict[str, Set[str]] = defaultdict(set)  # node -> dependencies

    def add_node(self, node: str):
        """添加节点"""
        self.nodes.add(node)

    def add_edge(self, from_node: str, to_node: str):
        """添加边"""
        self.nodes.add(from_node)
        self.nodes.add(to_node)
        self.edges[from_node].add(to_node)

    def get_dependents(self, node: str) -> Set[str]:
        """获取依赖该节点的所有节点"""
        dependents = set()
        for n, deps in self.edges.items():
            if node in deps:
                dependents.add(n)
        return dependents

    def get_dependencies(self, node: str) -> Set[str]:
        """获取该节点依赖的所有节点"""
        return self.edges.get(node, set())

    def find_circular_dependencies(self) -> List[List[str]]:
        """查找循环依赖"""
        cycles = []
        visited = set()
        rec_stack = set()
        path = []

        def dfs(node: str) -> Optional[List[str]]:
            visited.add(node)
            rec_stack.add(node)
            path.append(node)

            for neighbor in self.edges.get(node, []):
                if neighbor not in visited:
                    result = dfs(neighbor)
                    if result:
                        return result
                elif neighbor in rec_stack:
                    # 找到环
                    return path[path.index(neighbor):] + [neighbor]

            path.pop()
            rec_stack.remove(node)
            return None

        for node in self.nodes:
            if node not in visited:
                result = dfs(node)
                if result:
                    cycles.append(result)

        return cycles
```

### 13.4 代码模式识别

```python
class PatternRecognizer:
    """代码模式识别器"""

    def __init__(self):
        self.patterns = {
            "singleton": self._detect_singleton,
            "factory": self._detect_factory,
            "observer": self._detect_observer,
            "strategy": self._detect_strategy,
            "adapter": self._detect_adapter,
            "facade": self._detect_facade,
        }

    def recognize(self, files: List[Path]) -> PatternReport:
        """识别代码模式"""
        report = PatternReport()

        for file in files:
            content = file.read_text()

            for pattern_name, detector in self.patterns.items():
                if detector(content):
                    report.add_match(pattern_name, str(file))

        return report

    def _detect_singleton(self, content: str) -> bool:
        """检测单例模式"""
        # 检查是否有 __instance__ 或类似机制
        return bool(re.search(r'_instance\s*=', content))

    def _detect_factory(self, content: str) -> bool:
        """检测工厂模式"""
        # 检查是否有 create_* 方法
        return bool(re.search(r'def\s+create_\w+\(', content))

    def _detect_observer(self, content: str) -> bool:
        """检测观察者模式"""
        # 检查是否有 add_observer, notify 等方法
        has_observer = bool(re.search(r'add_observer|register_observer', content))
        has_notify = bool(re.search(r'notify|_observers', content))
        return has_observer and has_notify

    def _detect_strategy(self, content: str) -> bool:
        """检测策略模式"""
        # 检查是否有策略相关的接口/基类
        return bool(re.search(r'Strategy|Policy', content))


class AntiPatternRecognizer:
    """反模式识别器"""

    ANTI_PATTERNS = {
        "god_object": {
            "description": "上帝对象：一个类承担过多职责",
            "detector": lambda c: len(re.findall(r'def\s+\w+\(', c)) > 20
        },
        "long_method": {
            "description": "过长方法：方法超过 100 行",
            "detector": lambda c: any(len(m.group()) > 100 for m in re.finditer(r'def\s+\w+\([^)]*\):.*?(?=\ndef|\Z)', c, re.DOTALL))
        },
        "deep_nesting": {
            "description": "过度嵌套：嵌套超过 4 层",
            "detector": lambda c: max(self._count_nesting_depths(c)) > 4
        },
        "magic_numbers": {
            "description": "魔法数字：代码中直接使用数字字面量",
            "detector": lambda c: len(re.findall(r'\b\d{2,}\b', c)) > 5
        }
    }

    def recognize(self, files: List[Path]) -> AntiPatternReport:
        """识别反模式"""
        report = AntiPatternReport()

        for file in files:
            content = file.read_text()

            for pattern_name, info in self.ANTI_PATTERNS.items():
                if info["detector"](content):
                    report.add_violation(pattern_name, str(file), info["description"])

        return report

    def _count_nesting_depths(self, content: str) -> List[int]:
        """计算嵌套深度"""
        depths = []
        current_depth = 0
        max_depth = 0

        for line in content.split('\n'):
            stripped = line.lstrip()
            if stripped.startswith(('if', 'for', 'while', 'def', 'class')):
                current_depth += 1
                max_depth = max(max_depth, current_depth)
            elif stripped.startswith(('return', 'break', 'continue')) and current_depth > 0:
                depths.append(current_depth)
                current_depth = max(0, current_depth - 1)

        depths.append(current_depth)
        return depths if depths else [0]
```

### 13.5 语义地图生成

```python
@dataclass
class SemanticMap:
    """语义地图"""
    modules: List[Module]
    crossCuttingConcerns: List[CrossCuttingConcern]
    layerBoundaries: List[LayerBoundary]
    keyAbstractions: List[KeyAbstraction]


class SemanticMapGenerator:
    """语义地图生成器"""

    def generate(
        self,
        structure: StructureAnalysis,
        dependencies: DependencyGraph,
        patterns: PatternReport
    ) -> SemanticMap:
        """生成语义地图"""

        # 1. 识别模块
        modules = self._identify_modules(structure)

        # 2. 识别横切关注点
        crossCutting = self._identify_cross_cutting(dependencies)

        # 3. 识别分层边界
        layers = self._identify_layers(dependencies)

        # 4. 识别关键抽象
        abstractions = self._identify_key_abstractions(patterns)

        return SemanticMap(
            modules=modules,
            crossCuttingConcerns=crossCutting,
            layerBoundaries=layers,
            keyAbstractions=abstractions
        )

    def _identify_modules(self, structure: StructureAnalysis) -> List[Module]:
        """识别模块"""
        # 按功能分组
        modules = []

        # 常见模块模式
        module_patterns = {
            "auth": ["auth", "login", "user", "permission"],
            "api": ["api", "controller", "endpoint", "route"],
            "database": ["db", "repository", "model", "schema"],
            "service": ["service", "business", "logic"],
            "util": ["util", "helper", "common"],
        }

        for dir_path, stats in structure.directories.items():
            dir_name = dir_path.name.lower()

            for module_type, keywords in module_patterns.items():
                if any(kw in dir_name for kw in keywords):
                    modules.append(Module(
                        name=dir_name,
                        type=module_type,
                        path=dir_path,
                        file_count=stats.file_count,
                        lines=stats.total_lines
                    ))
                    break

        return modules

    def _identify_layers(self, dependencies: DependencyGraph) -> List[Layer]:
        """识别分层架构"""
        # 基于依赖关系的分层
        layers = []

        # 第 1 层：基础设施（没有对外依赖）
        layer1 = [n for n in dependencies.nodes if not dependencies.get_dependencies(n)]
        layers.append(Layer(name="Infrastructure", level=1, components=layer1))

        # 第 2 层：业务逻辑（只依赖基础设施）
        layer2_candidates = [
            n for n in dependencies.nodes
            if dependencies.get_dependencies(n) and
            all(dep in layer1 for dep in dependencies.get_dependencies(n))
        ]
        layers.append(Layer(name="Business Logic", level=2, components=layer2_candidates))

        # 第 3 层：应用服务（依赖业务逻辑）
        # ...

        return layers
```

### 13.6 可视化报告

```python
def generate_exploration_report(map: SemanticMap) -> str:
    """生成探索报告"""

    report = f"""# 代码库探索报告

## 概览

- **总模块数**: {len(map.modules)}
- **横切关注点数**: {len(map.crossCuttingConcerns)}
- **分层数**: {len(map.layerBoundaries)}
- **关键抽象数**: {len(map.keyAbstractions)}

## 模块列表

| 模块 | 类型 | 文件数 | 代码行数 |
|------|------|--------|----------|
"""

    for module in sorted(map.modules, key=lambda m: m.lines, reverse=True):
        report += f"| {module.name} | {module.type} | {module.file_count} | {module.lines} |\n"

    report += """
## 分层架构

```
"""

    for layer in sorted(map.layerBoundaries, key=lambda l: l.level):
        report += f"Layer {layer.level} ({layer.name}):\n"
        for component in layer.components[:5]:  # 只显示前5个
            report += f"  - {component}\n"
        if len(layer.components) > 5:
            report += f"  ... and {len(layer.components) - 5} more\n"
        report += "\n"

    report += """
```

## 关键抽象

"""

    for abstraction in map.keyAbstractions[:10]:
        report += f"""### {abstraction.name}

- **类型**: {abstraction.type}
- **位置**: {abstraction.location}
- **描述**: {abstraction.description}

"""

    return report
```
