### 18.1 团队应用概述

#### Claude Code 在团队中的价值

```
团队痛点：

1. 代码风格不统一
   - 每个人写法不同
   - Code Review 负担重

2. 知识传承困难
   - 老员工离职带走经验
   - 新人上手周期长

3. 代码质量不稳定
   - 依赖个人能力
   - 缺乏统一标准

4. 协作效率低
   - 接口对接不清晰
   - 文档不完善
```

```
Claude Code 解决方案：

1. 统一代码规范
   - 自动格式化
   - 风格检查

2. 知识固化
   - 团队共享 CLAUDE.md
   - 经验文档化

3. 质量保障
   - 自动审查
   - 安全扫描

4. 高效协作
   - 标准化接口
   - 自动生成文档
```

### 18.3 代码审查工作流

```python
class TeamCodeReviewWorkflow:
    """团队代码审查工作流"""

    def __init__(self):
        self.review_agent = CodeReviewAgent()
        self.security_agent = SecurityAuditAgent()
        self.quality_agent = QualityCheckAgent()

    async def run_review(self, pr: PullRequest) -> ReviewReport:
        """执行完整的代码审查"""
        report = ReviewReport(pr_id=pr.id)

        # 1. 获取代码变更
        diff = await pr.get_diff()

        # 2. 代码风格检查
        style_issues = await self.check_style(diff)
        report.add_issues("style", style_issues)

        # 3. 安全审查
        security_issues = await self.security_agent.audit(diff)
        report.add_issues("security", security_issues)

        # 4. 质量检查
        quality_issues = await self.quality_agent.check(diff)
        report.add_issues("quality", quality_issues)

        # 5. 生成审查报告
        report.generate_summary()

        return report


class CodeReviewAgent:
    """代码审查 Agent"""

    CHECKS = {
        "naming": self.check_naming,
        "complexity": self.check_complexity,
        "duplication": self.check_duplication,
        "documentation": self.check_documentation,
    }

    async def review(self, diff: Diff) -> List[ReviewComment]:
        """审查代码变更"""
        comments = []

        for check_name, check_func in self.CHECKS.items():
            issues = await check_func(diff)
            for issue in issues:
                comments.append(ReviewComment(
                    type=check_name,
                    severity=issue.severity,
                    file=issue.file,
                    line=issue.line,
                    message=issue.message,
                    suggestion=issue.suggestion
                ))

        return comments
```

class TeamKnowledgeBase:
    """团队知识库"""

    def __init__(self):
        self.documents = DocumentStore()
        self.architecture = ArchitectureMap()
        self.decisions = DecisionLog()

    async def add_decision(
        self,
        title: str,
        context: str,
        decision: str,
        rationale: str
    ):
        """记录架构决策"""
        decision = ArchitectureDecision(
            title=title,
            context=context,
            decision=decision,
            rationale=rationale,
            date=datetime.now(),
            author=get_current_user()
        )

        self.decisions.add(decision)

        # 更新架构图
        self.architecture.update_from_decision(decision)

    async def search(self, query: str) -> SearchResult:
        """搜索知识库"""
        results = []

        # 搜索文档
        docs = await self.documents.search(query)
        results.extend(docs)

        # 搜索决策
        decisions = await self.decisions.search(query)
        results.extend(decisions)

        return sorted(results, key=lambda x: x.relevance, reverse=True)


class ArchitectureDecision:
    """架构决策记录"""

    def __init__(
        self,
        title: str,
        context: str,
        decision: str,
        rationale: str,
        date: datetime,
        author: str
    ):
        self.title = title
        self.context = context
        self.decision = decision
        self.rationale = rationale
        self.date = date
        self.author = author
        self.consequences = []
        self.related_decisions = []

### 18.5 团队效率指标

```python
class TeamMetrics:
    """团队效率指标"""

    METRICS = {
        "code_quality": {
            "test_coverage": {"target": "> 80%"},
            "cyclomatic_complexity": {"target": "< 10"},
            "duplication_rate": {"target": "< 5%"},
        },
        "review_efficiency": {
            "avg_review_time": {"target": "< 24h"},
            "review_comments_per_pr": {"target": "> 3"},
            "approval_rate": {"target": "> 90%"},
        },
        "development_velocity": {
            "prs_per_week": {"target": "> 10"},
            "lead_time": {"target": "< 2 days"},
            "cycle_time": {"target": "< 3 days"},
        }
    }

    def calculate_score(self, actual: dict, target: dict) -> float:
        """计算指标得分"""
        if isinstance(target, dict):
            if "target" in target:
                return self._compare_value(actual, target["target"])
            else:
                return sum(
                    self.calculate_score(actual[k], target[k])
                    for k in target.keys()
                ) / len(target)
        return 1.0 if actual == target else 0.0

    def _compare_value(self, actual: str, target: str) -> float:
        """比较单个值"""
        # 解析目标值（如 "> 80%"）
        import re

        match = re.match(r'([<>=]+)\s*(\d+\.?\d*)', target)
        if not match:
            return 1.0 if actual == target else 0.0

        op, value = match.groups()
        actual_num = float(re.search(r'\d+\.?\d*', str(actual)).group())

        if op == ">":
            return 1.0 if actual_num > float(value) else 0.0
        elif op == "<":
            return 1.0 if actual_num < float(value) else 0.0
        elif op == ">=":
            return 1.0 if actual_num >= float(value) else 0.0
        elif op == "<=":
            return 1.0 if actual_num <= float(value) else 0.0
        else:
            return 1.0 if actual == target else 0.0
```

class OnboardingAutomation:
    """新成员入职自动化"""

    CHECKLIST = [
        "clone_repository",
        "install_dependencies",
        "setup_environment",
        "configure_editor",
        "run_tests",
        "read_architecture",
        "meet_team_members",
        "first_contribution"
    ]

    async def run_onboarding(self, user: User) -> OnboardingProgress:
        """执行入职流程"""
        progress = OnboardingProgress(user=user)

        for step_name in self.CHECKLIST:
            step = self._get_step(step_name)

            # 检查前置条件
            if not await step.check_prerequisites(user):
                progress.skip(step_name, "Prerequisites not met")
                continue

            # 执行步骤
            result = await step.execute(user)

            if result.success:
                progress.complete(step_name, result.output)
            else:
                progress.fail(step_name, result.error)

        return progress
