### 11.1 调试概述

#### 为什么 Skills 需要调试？

```
Skills 是复杂系统，容易出现问题的环节：

1. 触发阶段
   - 触发条件不匹配
   - 优先级冲突
   - 关键词覆盖

2. 解析阶段
   - 参数格式错误
   - 参数验证失败
   - 缺少必需参数

3. 执行阶段
   - 逻辑错误
   - 依赖缺失
   - 超时问题

4. 输出阶段
   - 输出格式错误
   - 结果不符合预期
   - 用户体验问题
```

#### 调试方法论

```
调试流程：

1. 复现问题
   - 确定触发条件
   - 准备测试输入
   - 观察实际输出

2. 分析原因
   - 查看日志
   - 分步执行
   - 检查状态

3. 定位问题
   - 缩小范围
   - 定位到具体步骤

4. 修复问题
   - 修改代码
   - 验证修复

5. 回归测试
   - 确保没有破坏其他功能
   - 测试边界情况
```

import logging
from datetime import datetime

class SkillLogger:
    """Skill 调试日志"""

    def __init__(self, skill_name: str, level: int = logging.DEBUG):
        self.skill_name = skill_name
        self.logger = logging.getLogger(f"skill.{skill_name}")
        self.logger.setLevel(level)

        # 添加文件处理器
        handler = logging.FileHandler(f"~/.claude/logs/skills/{skill_name}.log")
        formatter = logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)

    def debug(self, message: str, context: Dict = None):
        """调试日志"""
        msg = self._format_message(message, context)
        self.logger.debug(msg)

    def info(self, message: str, context: Dict = None):
        """信息日志"""
        msg = self._format_message(message, context)
        self.logger.info(msg)

    def error(self, message: str, context: Dict = None):
        """错误日志"""
        msg = self._format_message(message, context)
        self.logger.error(msg)

    def _format_message(self, message: str, context: Dict = None) -> str:
        """格式化消息"""
        parts = [f"[{self.skill_name}] {message}"]

        if context:
            parts.append(f"Context: {json.dumps(context, ensure_ascii=False)}")

        return " | ".join(parts)


class DebugContext:
    """调试上下文"""

    def __init__(self, skill_name: str):
        self.skill_name = skill_name
        self.logger = SkillLogger(skill_name)
        self.start_time = datetime.now()
        self.checkpoints = []

    def checkpoint(self, name: str, data: Any = None):
        """记录检查点"""
        elapsed = (datetime.now() - self.start_time).total_seconds()

        checkpoint = {
            "name": name,
            "elapsed": elapsed,
            "timestamp": datetime.now().isoformat(),
            "data": data
        }

        self.checkpoints.append(checkpoint)
        self.logger.debug(f"Checkpoint: {name}", {"elapsed": elapsed, "data": data})

        return checkpoint

    def get_trace(self) -> list:
        """获取追踪轨迹"""
        return self.checkpoints

# 问题诊断流程

class TriggerDiagnostics:
    """触发诊断"""

    def __init__(self):
        self.triggers = []

    def diagnose(self, user_input: str) -> Dict:
        """诊断触发问题"""
        results = {
            "input": user_input,
            "triggers_tested": [],
            "matched": None,
            "potential_issues": []
        }

        # 测试每个触发器
        for trigger in self.triggers:
            result = self._test_trigger(user_input, trigger)
            results["triggers_tested"].append(result)

            if result["matched"]:
                results["matched"] = trigger

        # 分析潜在问题
        if not results["matched"]:
            results["potential_issues"] = self._analyze_issues(user_input)

        return results

    def _test_trigger(self, user_input: str, trigger: Dict) -> Dict:
        """测试单个触发器"""
        trigger_text = trigger["trigger"].lower()
        user_input_lower = user_input.lower()

        matched = False
        match_type = None

        # 精确匹配
        if user_input_lower == trigger_text:
            matched = True
            match_type = "exact"

        # 前缀匹配
        elif user_input_lower.startswith(trigger_text):
            matched = True
            match_type = "prefix"

        # 包含匹配
        elif trigger_text in user_input_lower:
            matched = True
            match_type = "contains"

        return {
            "trigger": trigger["trigger"],
            "matched": matched,
            "match_type": match_type
        }

    def _analyze_issues(self, user_input: str) -> List[str]:
        """分析潜在问题"""
        issues = []

        # 检查是否以 / 开头（斜杠命令需要）
        if not user_input.startswith('/'):
            issues.append("可能需要以 / 开头（如 /skill-name）")

        # 检查拼写
        suggestions = self._suggest_corrections(user_input)
        if suggestions:
            issues.append(f"可能是想输入：{', '.join(suggestions)}")

        # 检查长度
        if len(user_input) < 3:
            issues.append("输入太短，可能无法匹配")

        return issues

    def _suggest_corrections(self, user_input: str) -> List[str]:
        """建议更正"""
        # 简单的拼写纠正逻辑
        suggestions = []

        for trigger in self.triggers:
            distance = self._levenshtein_distance(
                user_input.lower(),
                trigger["trigger"].lower()
            )

            if distance <= 2:  # 编辑距离小于等于2
                suggestions.append(trigger["trigger"])

        return suggestions[:3]

    def _levenshtein_distance(self, s1: str, s2: str) -> int:
        """计算编辑距离"""
        if len(s1) < len(s2):
            return self._levenshtein_distance(s2, s1)

        if len(s2) == 0:
            return len(s1)

        previous_row = range(len(s2) + 1)

        for i, c1 in enumerate(s1):
            current_row = [i + 1]

            for j, c2 in enumerate(s2):
                insertions = previous_row[j + 1] + 1
                deletions = current_row[j] + 1
                substitutions = previous_row[j] + (c1 != c2)
                current_row.append(min(insertions, deletions, substitutions))

            previous_row = current_row

        return previous_row[-1]

class ConversationAnalyzer:
    """对话流程分析"""

    def __init__(self):
        self.turns = []

    def add_turn(self, role: str, content: str, metadata: Dict = None):
        """添加对话轮次"""
        self.turns.append({
            "role": role,
            "content": content,
            "timestamp": datetime.now(),
            "metadata": metadata or {}
        })

    def analyze_flow(self) -> Dict:
        """分析对话流程"""
        return {
            "total_turns": len(self.turns),
            "user_turns": self._count_turns("user"),
            "assistant_turns": self._count_turns("assistant"),
            "avg_turn_length": self._avg_turn_length(),
            "topics": self._extract_topics(),
            "issues": self._detect_issues()
        }

    def _count_turns(self, role: str) -> int:
        """计算特定角色的轮次"""
        return sum(1 for t in self.turns if t["role"] == role)

    def _avg_turn_length(self) -> float:
        """平均轮次长度"""
        if not self.turns:
            return 0

        total_length = sum(len(t["content"]) for t in self.turns)
        return total_length / len(self.turns)

    def _extract_topics(self) -> List[str]:
        """提取对话主题"""
        # 简单的主题提取
        topics = []

        keywords = {
            "git": ["git", "commit", "branch", "merge"],
            "debug": ["bug", "error", "fix", "issue"],
            "review": ["review", "code", "check"],
            "deploy": ["deploy", "release", "push", "production"]
        }

        all_content = " ".join(t["content"].lower() for t in self.turns)

        for topic, kws in keywords.items():
            if any(kw in all_content for kw in kws):
                topics.append(topic)

        return topics

    def _detect_issues(self) -> List[Dict]:
        """检测对话问题"""
        issues = []

        # 检测：用户重复同一问题
        if self._has_repeated_questions():
            issues.append({
                "type": "repeated_question",
                "severity": "medium",
                "message": "用户可能没有得到满意的答案"
            })

        # 检测：助手回复过长
        long_responses = self._get_long_responses()
        if long_responses:
            issues.append({
                "type": "long_responses",
                "severity": "low",
                "message": f"有 {len(long_responses)} 个较长的回复"
            })

        # 检测：上下文丢失
        if self._has_context_gaps():
            issues.append({
                "type": "context_gap",
                "severity": "high",
                "message": "可能存在上下文丢失"
            })

        return issues

    def _has_repeated_questions(self) -> bool:
        """检测重复问题"""
        user_turns = [t["content"] for t in self.turns if t["role"] == "user"]

        # 简单检测：连续两个相同的问题
        for i in range(len(user_turns) - 1):
            if user_turns[i] == user_turns[i + 1]:
                return True

        return False

    def _get_long_responses(self) -> List[Dict]:
        """获取过长回复"""
        threshold = 2000  # 2000 字符
        long_responses = []

        for turn in self.turns:
            if turn["role"] == "assistant" and len(turn["content"]) > threshold:
                long_responses.append({
                    "length": len(turn["content"]),
                    "preview": turn["content"][:100] + "..."
                })

        return long_responses

    def _has_context_gaps(self) -> bool:
        """检测上下文缺失"""

class SkillDebugger:
    """Skill 调试器"""

    def __init__(self, skill: "Skill"):
        self.skill = skill
        self.breakpoints = {}
        self.watch_variables = {}

    def set_breakpoint(self, step_name: str):
        """设置断点"""
        self.breakpoints[step_name] = True

    def remove_breakpoint(self, step_name: str):
        """移除断点"""
        if step_name in self.breakpoints:
            del self.breakpoints[step_name]

    def watch(self, variable_name: str):
        """监视变量"""
        self.watch_variables[variable_name] = []

    async def run(self, input_data: Any) -> Any:
        """调试运行"""
        context = {"input": input_data, "_debugger": self}

        for step in self.skill.steps:
            # 检查断点
            if step["name"] in self.breakpoints:
                await self._hit_breakpoint(step, context)

            # 记录变量
            for var_name in self.watch_variables:
                if var_name in context:
                    self.watch_variables[var_name].append(context[var_name])

            # 执行步骤
            result = await self._execute_step(step, context)
            context[step["name"]] = result

        return context.get("result")

    async def _hit_breakpoint(self, step: Dict, context: Dict):
        """断点命中"""
        print(f"Breakpoint hit at: {step['name']}")
        print(f"Context: {json.dumps(context, indent=2)}")

        # 等待用户命令
        while True:
            cmd = input("(debug) ")

            if cmd == "continue":
                break
            elif cmd == "inspect":
                print(json.dumps(context, indent=2))
            elif cmd.startswith("watch "):
                var = cmd.split()[1]
                print(f"{var} = {context.get(var)}")
            elif cmd == "quit":
                raise InterruptedError("Debug session ended")

class PerformanceProfiler:
    """性能分析器"""

    def __init__(self):
        self.profiles = {}

    def profile(self, skill_name: str):
        """性能分析装饰器"""
        def decorator(func):
            async def wrapper(*args, **kwargs):
                start_time = time.time()
                start_memory = self._get_memory_usage()

                try:
                    result = await func(*args, **kwargs)
                    return result
                finally:
                    elapsed = time.time() - start_time
                    memory_delta = self._get_memory_usage() - start_memory

                    self.profiles[skill_name] = {
                        "elapsed": elapsed,
                        "memory_delta": memory_delta,
                        "timestamp": datetime.now()
                    }

            return wrapper
        return decorator

    def _get_memory_usage(self) -> float:
        """获取内存使用"""
        import psutil
        process = psutil.Process()
        return process.memory_info().rss / 1024 / 1024  # MB

    def get_report(self) -> str:
        """获取性能报告"""
        report = "# 性能分析报告\n\n"

        for skill_name, profile in sorted(
            self.profiles.items(),
            key=lambda x: x[1]["elapsed"],
            reverse=True
        ):
            report += f"## {skill_name}\n"
            report += f"- 执行时间: {profile['elapsed']:.3f}s\n"
            report += f"- 内存变化: {profile['memory_delta']:.2f}MB\n"
            report += f"- 时间戳: {profile['timestamp']}\n\n"

        return report
