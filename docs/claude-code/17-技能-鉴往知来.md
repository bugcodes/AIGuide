### 15.1 测试覆盖基础

#### 什么是测试覆盖？

```
测试覆盖 = 代码被执行的比例

类型：
1. 行覆盖（Line Coverage）
   - 代码行被执行的比例

2. 分支覆盖（Branch Coverage）
   - if/else 路径被执行的比例

3. 函数覆盖（Function Coverage）
   - 函数被调用的比例

4. 路径覆盖（Path Coverage）
   - 所有可能执行路径的比例
```

#### 覆盖率指标计算

```python
@dataclass
class CoverageMetrics:
    """覆盖指标"""
    total_lines: int = 0
    covered_lines: int = 0
    total_branches: int = 0
    covered_branches: int = 0
    total_functions: int = 0
    covered_functions: int = 0

    @property
    def line_coverage(self) -> float:
        """行覆盖率"""
        if self.total_lines == 0:
            return 0.0
        return (self.covered_lines / self.total_lines) * 100

    @property
    def branch_coverage(self) -> float:
        """分支覆盖率"""
        if self.total_branches == 0:
            return 0.0
        return (self.covered_branches / self.total_branches) * 100

    @property
    def function_coverage(self) -> float:
        """函数覆盖率"""
        if self.total_functions == 0:
            return 0.0
        return (self.covered_functions / self.total_functions) * 100


class CoverageCalculator:
    """覆盖率计算器"""

    def __init__(self):
        self.executed_lines: Set[int] = set()
        self.executed_branches: Set[Tuple[int, str]] = set()
        self.executed_functions: Set[str] = set()

    def analyze_file(self, source_file: Path, trace_file: Path) -> CoverageMetrics:
        """分析单个文件的覆盖率"""
        source = source_file.read_text()
        lines = source.split('\n')

        # 解析 trace 文件
        trace_data = self._parse_trace(trace_file)

        metrics = CoverageMetrics()

        # 统计总行数（排除空行和注释）
        metrics.total_lines = self._count_code_lines(lines)

        # 统计执行的行数
        metrics.covered_lines = self._count_covered_lines(lines, trace_data)

        # 统计分支数
        metrics.total_branches = self._count_branches(lines)
        metrics.covered_branches = self._count_covered_branches(lines, trace_data)

        # 统计函数数
        metrics.total_functions = self._count_functions(lines)
        metrics.covered_functions = self._count_covered_functions(lines, trace_data)

        return metrics

    def _count_code_lines(self, lines: List[str]) -> int:
        """统计代码行数（排除空行和注释）"""
        count = 0
        in_multiline_comment = False

        for line in lines:
            stripped = line.strip()

            # 多行注释
            if stripped.startswith('"""') or stripped.startswith("'''"):
                in_multiline_comment = not in_multiline_comment
                continue

            if in_multiline_comment:
                continue

            # 单行注释
            if stripped.startswith('#') or stripped.startswith('//'):
                continue

            # 空行
            if not stripped:
                continue

            count += 1

        return count

    def _count_branches(self, lines: List[str]) -> int:
        """统计分支数"""
        branch_keywords = ['if', 'elif', 'else', 'for', 'while', 'try', 'except', 'finally']

        count = 0
        for line in lines:
            for keyword in branch_keywords:
                if keyword in line:
                    count += 1
                    break

        return count
```

### 15.2 测试覆盖分析

```python
class CoverageAnalyzer:
    """测试覆盖分析器"""

    def __init__(self):
        self.metrics: Dict[str, CoverageMetrics] = {}

    def analyze_coverage(self, project_path: Path) -> CoverageReport:
        """分析项目覆盖情况"""
        report = CoverageReport()

        # 收集所有源文件和对应的 trace
        source_files = list(project_path.rglob("*.py"))
        trace_files = list(project_path.rglob(".coverage"))

        for source_file in source_files:
            # 查找对应的 trace
            trace_file = self._find_trace(source_file, trace_files)

            if trace_file:
                metrics = self._analyze_file(source_file, trace_file)
                self.metrics[str(source_file)] = metrics

                report.add_file_metrics(source_file, metrics)

        return report

    def _find_trace(self, source_file: Path, trace_files: List[Path]) -> Optional[Path]:
        """查找对应的 trace 文件"""
        # 简化：假设 trace 文件与源文件同名
        base = source_file.stem
        for trace in trace_files:
            if base in str(trace):
                return trace
        return None

    def get_uncovered_lines(self, file_path: str) -> List[int]:
        """获取未覆盖的行号"""
        if file_path not in self.metrics:
            return []

        metrics = self.metrics[file_path]
        source = Path(file_path).read_text().split('\n')

        uncovered = []
        for i, line in enumerate(source, 1):
            if self._is_code_line(line) and i not in self.executed_lines:
                uncovered.append(i)

        return uncovered

    def _is_code_line(self, line: str) -> bool:
        """判断是否是代码行"""
        stripped = line.strip()
        return bool(stripped) and not stripped.startswith('#')


class CoverageReport:
    """覆盖报告"""

    def __init__(self):
        self.files: Dict[str, CoverageMetrics] = {}

    def add_file_metrics(self, file: Path, metrics: CoverageMetrics):
        """添加文件覆盖指标"""
        self.files[str(file)] = metrics

    def get_summary(self) -> Dict:
        """获取汇总信息"""
        total_lines = sum(m.total_lines for m in self.files.values())
        covered_lines = sum(m.covered_lines for m in self.files.values())

        return {
            "total_files": len(self.files),
            "total_lines": total_lines,
            "covered_lines": covered_lines,
            "line_coverage": (covered_lines / total_lines * 100) if total_lines > 0 else 0
        }

    def generate_report(self) -> str:
        """生成报告"""
        summary = self.get_summary()

        report = f"""# 测试覆盖报告

## 概览

- **总文件数**: {summary['total_files']}
- **总代码行**: {summary['total_lines']}
- **覆盖行数**: {summary['covered_lines']}
- **行覆盖率**: {summary['line_coverage']:.1f}%

## 文件详情

| 文件 | 总行数 | 覆盖行数 | 覆盖率 |
|------|--------|----------|--------|
"""

        for file_path, metrics in sorted(
            self.files.items(),
            key=lambda x: x[1].line_coverage
        ):
            report += f"| {Path(file_path).name} | {metrics.total_lines} | {metrics.covered_lines} | {metrics.line_coverage:.1f}% |\n"

        return report
```

### 15.3 代码推荐系统

```python
class CodeRecommender:
    """代码推荐器"""

    def __init__(self, coverage_analyzer: CoverageAnalyzer = None):
        self.coverage_analyzer = coverage_analyzer
        self.patterns = PatternDatabase()

    def suggest_tests(self, file_path: str) -> List[TestSuggestion]:
        """基于覆盖分析建议测试"""
        suggestions = []

        if not self.coverage_analyzer:
            return suggestions

        uncovered_lines = self.coverage_analyzer.get_uncovered_lines(file_path)

        for line_no in uncovered_lines:
            suggestion = self._analyze_uncovered_line(file_path, line_no)
            if suggestion:
                suggestions.append(suggestion)

        return suggestions

    def _analyze_uncovered_line(self, file_path: str, line_no: int) -> Optional[TestSuggestion]:
        """分析未覆盖的行"""
        source = Path(file_path).read_text()
        lines = source.split('\n')

        if line_no > len(lines):
            return None

        line = lines[line_no - 1]

        # 分析代码类型
        if 'def ' in line:
            return self._suggest_function_test(file_path, line_no, line)
        elif 'if ' in line:
            return self._suggest_branch_test(file_path, line_no, line)
        elif 'for ' in line or 'while ' in line:
            return self._suggest_loop_test(file_path, line_no, line)
        elif 'try:' in line:
            return self._suggest_exception_test(file_path, line_no, line)

        return None

    def _suggest_function_test(
        self,
        file_path: str,
        line_no: int,
        line: str
    ) -> TestSuggestion:
        """建议函数测试"""
        func_name = line.split('def ')[1].split('(')[0]

        return TestSuggestion(
            type="function",
            target=f"{file_path}:{line_no}",
            title=f"为函数 '{func_name}' 添加测试",
            description=f"函数 {func_name} 在第 {line_no} 行定义但未被测试",
            priority="high",
            test_template=self._generate_function_test(func_name)
        )

    def _generate_function_test(self, func_name: str) -> str:
        """生成函数测试模板"""
        return f'''def test_{func_name}():
    """测试 {func_name}"""
    # TODO: 添加测试逻辑
    pass
'''

    def suggest_code_completion(
        self,
        context: str,
        cursor_position: int
    ) -> List[CodeCompletion]:
        """建议代码补全"""
        completions = []

        # 分析上下文
        context_analysis = self._analyze_context(context, cursor_position)

        # 基于上下文的推荐
        if context_analysis['last_word']:
            completions.extend(self._suggest_by_pattern(context_analysis))

        # 基于语义的推荐
        completions.extend(self._suggest_by_semantics(context_analysis))

        return completions[:5]  # 返回前5个

    def _analyze_context(self, context: str, cursor: int) -> Dict:
        """分析上下文"""
        before_cursor = context[:cursor]
        after_cursor = context[cursor:]

        # 获取前一个单词
        words = before_cursor.split()
        last_word = words[-1] if words else ""

        # 获取前一行
        lines = before_cursor.split('\n')
        last_line = lines[-1] if lines else ""

        return {
            "last_word": last_word,
            "last_line": last_line,
            "before_cursor": before_cursor,
            "after_cursor": after_cursor,
            "indentation": len(last_line) - len(last_line.lstrip())
        }
```

### 15.4 智能代码生成

```python
class IntelligentCodeGenerator:
    """智能代码生成器"""

    def __init__(self):
        self.template_engine = TemplateEngine()
        self.context_analyzer = ContextAnalyzer()

    async def generate_from_context(
        self,
        file_path: str,
        cursor_line: int,
        instruction: str
    ) -> str:
        """基于上下文生成代码"""
        # 1. 分析上下文
        context = await self.context_analyzer.analyze(
            file_path,
            cursor_line
        )

        # 2. 选择合适的模板
        template = self._select_template(instruction, context)

        # 3. 填充模板
        code = self.template_engine.render(template, context)

        return code

    def _select_template(self, instruction: str, context: Dict) -> str:
        """选择模板"""
        if 'add test' in instruction.lower():
            return self._get_test_template(context)
        elif 'add error handling' in instruction.lower():
            return self._get_error_handling_template(context)
        elif 'add logging' in instruction.lower():
            return self._get_logging_template(context)
        else:
            return self._get_generic_template(context)


class ContextAnalyzer:
    """上下文分析器"""

    async def analyze(self, file_path: str, cursor_line: int) -> Dict:
        """分析上下文"""
        source = Path(file_path).read_text()
        lines = source.split('\n')

        # 获取周围的代码
        start = max(0, cursor_line - 10)
        end = min(len(lines), cursor_line + 10)
        surrounding = lines[start:end]

        # 分析代码类型
        code_type = self._detect_code_type('\n'.join(surrounding))

        # 分析函数/类定义
        function_def = self._find_enclosing_function(lines, cursor_line)
        class_def = self._find_enclosing_class(lines, cursor_line)

        return {
            "file_path": file_path,
            "cursor_line": cursor_line,
            "surrounding_code": '\n'.join(surrounding),
            "code_type": code_type,
            "enclosing_function": function_def,
            "enclosing_class": class_def,
            "imports": self._extract_imports(lines)
        }
```

### 15.5 覆盖驱动开发

```python
class CoverageDrivenDevelopment:
    """覆盖驱动开发"""

    def __init__(self):
        self.coverage_analyzer = CoverageAnalyzer()
        self.recommender = CodeRecommender()

    async def run_cdd_cycle(self, project_path: Path) -> CDDReport:
        """
        执行覆盖驱动开发周期

        流程：
        1. 运行测试，获取覆盖报告
        2. 识别未覆盖的代码
        3. 为未覆盖代码生成测试建议
        4. 生成代码补全建议
        5. 输出改进建议
        """
        report = CDDReport()

        # 1. 分析覆盖
        coverage = self.coverage_analyzer.analyze_coverage(project_path)
        report.coverage = coverage

        # 2. 识别未覆盖文件
        low_coverage_files = [
            (f, m) for f, m in coverage.files.items()
            if m.line_coverage < 80
        ]
        report.low_coverage_files = low_coverage_files

        # 3. 生成改进建议
        suggestions = []
        for file_path, metrics in low_coverage_files:
            suggestions.extend(
                self.recommender.suggest_tests(file_path)
            )

        report.suggestions = suggestions

        # 4. 生成优先级
        report.priority_files = self._prioritize_files(low_coverage_files)

        return report

    def _prioritize_files(
        self,
        files: List[Tuple[str, CoverageMetrics]]
    ) -> List[Dict]:
        """优先级排序"""
        prioritized = []

        for file_path, metrics in files:
            # 计算优先级分数
            score = (
                (100 - metrics.line_coverage) * 0.4 +  # 覆盖率缺口
                metrics.total_lines * 0.001 * 0.3 +    # 文件大小
                self._calculate_complexity_score(file_path) * 0.3  # 复杂度
            )

            prioritized.append({
                "file": file_path,
                "score": score,
                "current_coverage": metrics.line_coverage,
                "lines_to_cover": metrics.total_lines - metrics.covered_lines
            })

        return sorted(prioritized, key=lambda x: x["score"], reverse=True)


@dataclass
class CDDReport:
    """覆盖驱动开发报告"""
    coverage: CoverageReport = None
    low_coverage_files: List = field(default_factory=list)
    suggestions: List = field(default_factory=list)
    priority_files: List = field(default_factory=list)
```

### 15.6 Claude Code 集成

```python
class ClaudeCodeIntegration:
    """Claude Code 集成"""

    def __init__(self):
        self.cdd = CoverageDrivenDevelopment()

    async def run_with_coverage_guidance(self, project_path: Path) -> str:
        """运行带有覆盖指导的工作流"""
        suggestions = []

        # 1. 获取覆盖驱动开发建议
        report = await self.cdd.run_cdd_cycle(project_path)

        # 2. 生成改进建议
        if report.priority_files:
            suggestions.append("# 优先改进文件\n")

            for item in report.priority_files[:5]:
                suggestions.append(f"""
## {Path(item['file']).name}
- 当前覆盖率：{item['current_coverage']:.1f}%
- 待覆盖行数：{item['lines_to_cover']}
- 优先级分数：{item['score']:.2f}
""")

        # 3. 生成测试建议
        if report.suggestions:
            suggestions.append("# 测试建议\n")

            for suggestion in report.suggestions[:10]:
                suggestions.append(f"""
### {suggestion.title}

**位置**: {suggestion.target}
**优先级**: {suggestion.priority}
**描述**: {suggestion.description}

**测试模板**:
```python
{suggestion.test_template}
```
""")

        return "\n".join(suggestions)
```
